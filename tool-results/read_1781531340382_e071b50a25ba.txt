   216→   216→function parseCognitiveJSON(responseText: string): CognitiveResult {
   217→   217→  // Extract JSON from response (handle markdown code blocks)
   218→   218→  const jsonMatch = responseText.match(/\{[\s\S]*\}/)
   219→   219→  if (jsonMatch) {
   220→   220→    try {
   221→   221→      const parsed = JSON.parse(jsonMatch[0])
   222→   222→      return {
   223→   223→        summary: typeof parsed.summary === 'string' ? parsed.summary.trim() : '',
   224→   224→        deepInsight: typeof parsed.deep_insight === 'string' ? parsed.deep_insight.trim() : '',
   225→   225→        tags: Array.isArray(parsed.tags)
   226→   226→          ? parsed.tags
   227→   227→              .filter((t: unknown) => typeof t === 'string')
   228→   228→              .map((t: string) => t.toLowerCase().trim())
   229→   229→              .filter(Boolean)
   230→   230→              .slice(0, 5)
   231→   231→          : [],
   232→   232→      }
   233→   233→    } catch {
   234→   234→      // JSON parse failed — continue to regex extraction
   235→   235→    }
   236→   236→  }
   237→   237→
   238→   238→  // Fallback: try to extract individual fields via regex
   239→   239→  const summaryMatch = responseText.match(/"summary"\s*:\s*"([^"]*)"/)
   240→   240→  const insightMatch = responseText.match(/"deep_insight"\s*:\s*"([^"]*)"/)
   241→   241→
   242→   242→  return {
   243→   243→    summary: summaryMatch?.[1]?.trim() || '',
   244→   244→    deepInsight: insightMatch?.[1]?.trim() || '',
   245→   245→    tags: [],
   246→   246→  }
   247→   247→}

   270→   270→async function matchToExistingCollections(
   271→   271→  memoryId: string,
   272→   272→  memoryTags: string[],
   273→   273→  userId: string
   274→   274→): Promise<void> {
   275→   275→  try {
   276→   276→    const supabase = await getSupabaseServer()
   277→   277→
   278→   278→    // Fetch all user's collections
   279→   279→    const { data: userCollections, error: collError } = await supabase
   280→   280→      .from('collections')
   281→   281→      .select('id, name')
   282→   282→      .eq('user_id', userId)
   283→   283→
   284→   284→    if (collError || !userCollections) return
   285→   285→
   286→   286→    for (const collection of userCollections) {
   287→   287→      const collNameLower = collection.name.toLowerCase()
   288→   288→
   289→   289→      // Check if any of the memory's tags match this collection's name
   290→   290→      const tagMatches = memoryTags.some(
   291→   291→        (tag) =>
   292→   292→          collNameLower.includes(tag) ||
   293→   293→          tag.includes(collNameLower) ||
   294→   294→          levenshteinSimilarity(collNameLower, tag) > 0.6
   295→   295→      )
   296→   296→
   297→   297→      if (tagMatches) {
   298→   298→        // Check if this junction already exists
   299→   299→        const { data: existingJunction } = await supabase
   300→   300→          .from('memory_collections')
   301→   301→          .select('id')
   302→   302→          .eq('memory_id', memoryId)
   303→   303→          .eq('collection_id', collection.id)
   304→   304→          .limit(1)
   305→   305→
   306→   306→        if (!existingJunction || existingJunction.length === 0) {
   307→   307→          await supabase.from('memory_collections').insert({
   308→   308→            memory_id: memoryId,
   309→   309→            collection_id: collection.id,
   310→   310→          })
   311→   311→          console.log(`[capture] Auto-assigned memory ${memoryId} to collection "${collection.name}"`)
   312→   312→        }
   313→   313→      }
   314→   314→    }
   315→   315→  } catch (err) {
   316→   316→    console.error('[capture] Collection matching error:', err instanceof Error ? err.message : 'Unknown')
   317→   317→  }
   318→   318→}

   320→   320→async function checkAutoCollectionRule(userId: string): Promise<void> {
   321→   321→  try {
   322→   322→    const supabase = await getSupabaseServer()
   323→   323→
   324→   324→    // Fetch all user's memories that have tags
   325→   325→    const { data: allMemories, error: memError } = await supabase
   326→   326→      .from('memories')
   327→   327→      .select('id, tags')
   328→   328→      .eq('user_id', userId)
   329→   329→      .neq('tags', '')
   330→   330→
   331→   331→    if (memError || !allMemories || allMemories.length < 10) return
   332→   332→
   333→   333→    // Fetch all existing junction rows to determine which memories are collected
   334→   334→    const { data: junctions } = await supabase
   335→   335→      .from('memory_collections')
   336→   336→      .select('memory_id')
   337→   337→
   338→   338→    const collectedMemoryIds = new Set((junctions || []).map((j) => j.memory_id))
   339→   339→
   340→   340→    // Filter to uncollected memories only
   341→   341→    const uncollectedMemories = allMemories.filter((m) => !collectedMemoryIds.has(m.id))
   342→   342→
   343→   343→    if (uncollectedMemories.length < 10) return
   344→   344→
   345→   345→    // Build tag frequency map from uncollected memories
   346→   346→    const tagFrequency: Record<string, string[]> = {}
   347→   347→    for (const mem of uncollectedMemories) {
   348→   348→      const memTags = (mem.tags as string)
   349→   349→        .split(',')
   350→   350→        .map((t) => t.trim().toLowerCase())
   351→   351→        .filter(Boolean)
   352→   352→      for (const tag of memTags) {
   353→   353→        if (!tagFrequency[tag]) tagFrequency[tag] = []
   354→   354→        tagFrequency[tag].push(mem.id)
   355→   355→      }
   356→   356→    }
   357→   357→
   358→   358→    // Fetch existing collection names to avoid duplicates
   359→   359→    const { data: existingCollections } = await supabase
   360→   360→      .from('collections')
   361→   361→      .select('name')
   362→   362→      .eq('user_id', userId)
   363→   363→
   364→   364→    const existingNames = new Set(
   365→   365→      (existingCollections || []).map((c) => c.name.toLowerCase())
   366→   366→    )
   367→   367→
   368→   368→    const clusterColors = [
   369→   369→      '#6D597A', '#B56576', '#355070', '#EA526F',
   370→   370→      '#23B5D3', '#7B2D8E', '#F18F01', '#C73E1D',
   371→   371→      '#2D936C', '#566E3D', '#8B5CF6', '#EC4899',
   372→   372→    ]
   373→   373→    const { count: collectionCount } = await supabase
   374→   374→      .from('collections')
   375→   375→      .select('*', { count: 'exact', head: true })
   376→   376→      .eq('user_id', userId)
   377→   377→    let colorIdx = collectionCount || 0
   378→   378→
   379→   379→    // Find tag clusters that hit 10+ memories
   380→   380→    for (const [tag, memoryIds] of Object.entries(tagFrequency)) {
   381→   381→      if (memoryIds.length >= 10) {
   382→   382→        const collectionName = tag.charAt(0).toUpperCase() + tag.slice(1)
   383→   383→
   384→   384→        // Skip if a collection with this name already exists
   385→   385→        if (existingNames.has(collectionName.toLowerCase())) continue
   386→   386→
   387→   387→        const color = clusterColors[colorIdx % clusterColors.length]
   388→   388→        colorIdx++
   389→   389→
   390→   390→        // Create the new collection
   391→   391→        const { data: newCollection, error: createError } = await supabase
   392→   392→          .from('collections')
   393→   393→          .insert({
   394→   394→            user_id: userId,
   395→   395→            name: collectionName,
   396→   396→            icon: getTagIcon(tag),
   397→   397→            color,
   398→   398→          })
   399→   399→          .select('id')
   400→   400→          .single()
   401→   401→
   402→   402→        if (createError || !newCollection) {
   403→   403→          console.error('[capture] Auto-collection creation failed:', createError?.message)
   404→   404→          continue
   405→   405→        }
   406→   406→
   407→   407→        // Batch-assign all matching memories to this new collection
   408→   408→        const junctionInserts = memoryIds.slice(0, 100).map((memId) => ({
   409→   409→          memory_id: memId,
   410→   410→          collection_id: newCollection.id,
   411→   411→        }))
   412→   412→
   413→   413→        const { error: junctionError } = await supabase
   414→   414→          .from('memory_collections')
   415→   415→          .insert(junctionInserts)
   416→   416→
   417→   417→        if (junctionError) {
   418→   418→          console.error('[capture] Junction insert failed:', junctionError.message)
   419→   419→        } else {
   420→   420→          console.log(`[capture] Auto-created collection "${collectionName}" with ${memoryIds.length} memories`)
   421→   421→          existingNames.add(collectionName.toLowerCase())
   422→   422→        }
   423→   423→      }
   424→   424→    }
   425→   425→  } catch (err) {
   426→   426→    console.error('[capture] Auto-collection rule error:', err instanceof Error ? err.message : 'Unknown')

   451→   451→// ─── Levenshtein similarity for fuzzy tag matching ─────────────────
   452→   452→function levenshteinSimilarity(a: string, b: string): number {
   453→   453→  if (a === b) return 1
   454→   454→  if (!a.length || !b.length) return 0
   455→   455→
   456→   456→  const matrix: number[][] = []
   457→   457→  for (let i = 0; i <= b.length; i++) matrix[i] = [i]
   458→   458→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j
   459→   459→
   460→   460→  for (let i = 1; i <= b.length; i++) {
   461→   461→    for (let j = 1; j <= a.length; j++) {
   462→   462→      const cost = b[i - 1] === a[j - 1] ? 0 : 1
   463→   463→      matrix[i][j] = Math.min(
   464→   464→        matrix[i - 1][j] + 1,
   465→   465→        matrix[i][j - 1] + 1,
   466→   466→        matrix[i - 1][j - 1] + cost
   467→   467→      )
   468→   468→    }
   469→   469→  }
   470→   470→
   471→   471→  const maxLen = Math.max(a.length, b.length)
   472→   472→  return 1 - matrix[b.length][a.length] / maxLen
   473→   473→}
