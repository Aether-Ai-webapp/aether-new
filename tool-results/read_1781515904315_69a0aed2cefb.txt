   114→   114→// ─── Smart Token Saver: Local, free, instant tagging ──────────────────
   115→   115→function getLocalTags(text: string): string[] {
   116→   116→  const lower = text.toLowerCase()
   117→   117→  const tags: string[] = []
   118→   118→
   119→   119→  // URL detection → 'link'
   120→   120→  if (/https?:\/\//.test(lower)) {
   121→   121→    tags.push('link')
   122→   122→  }
   123→   123→
   124→   124→  // Task words → 'task'
   125→   125→  if (/\b(todo|remind|need to|must|buy)\b/.test(lower)) {
   126→   126→    tags.push('task')
   127→   127→  }
   128→   128→
   129→   129→  // Person/personal words → 'personal'
   130→   130→  if (/\b(i |me |with |call|email)\b/.test(lower)) {
   131→   131→    tags.push('personal')
   132→   132→  }
   133→   133→
   134→   134→  // Question/idea words → 'idea'
   135→   135→  if (/\b(what|how|why|who)\b/.test(lower)) {
   136→   136→    tags.push('idea')
   137→   137→  }
   138→   138→
   139→   139→  return tags.slice(0, 5)
   140→   140→}

   142→   142→// ─── Extended keyword-based auto-tagging (richer, for AI fallback) ───
   143→   143→function autoGenerateTags(content: string, title: string): string[] {
   144→   144→  const text = `${title} ${content}`.toLowerCase()
   145→   145→  const tagMap: Record<string, string[]> = {
   146→   146→    'work': ['meeting', 'project', 'deadline', 'client', 'office', 'team', 'q1', 'q2', 'q3', 'q4', 'quarterly', 'strategy'],
   147→   147→    'personal': ['routine', 'morning', 'exercise', 'meditation', 'journal', 'habit'],
   148→   148→    'travel': ['trip', 'itinerary', 'flight', 'hotel', 'visit', 'tokyo', 'paris', 'destination'],
   149→   149→    'learning': ['learn', 'study', 'course', 'tutorial', 'book', 'read', 'article'],
   150→   150→    'code': ['code', 'programming', 'react', 'javascript', 'typescript', 'api', 'bug', 'feature', 'css', 'html', 'framework'],
   151→   151→    'design': ['design', 'ui', 'ux', 'layout', 'color', 'font', 'figma', 'wireframe'],
   152→   152→    'ai': ['ai', 'machine learning', 'neural', 'model', 'gpt', 'gemini', 'llm', 'chatbot'],
   153→   153→    'recipe': ['recipe', 'cook', 'bake', 'ingredient', 'food', 'meal', 'breakfast', 'dinner'],
   154→   154→    'idea': ['idea', 'concept', 'brainstorm', 'innovative', 'startup', 'prototype'],
   155→   155→    'finance': ['budget', 'expense', 'invest', 'savings', 'money', 'cost'],
   156→   156→  }
   157→   157→
   158→   158→  const tags: string[] = []
   159→   159→  for (const [tag, keywords] of Object.entries(tagMap)) {
   160→   160→    if (keywords.some(kw => text.includes(kw))) {
   161→   161→      tags.push(tag)
   162→   162→    }
   163→   163→  }
   164→   164→  return tags.slice(0, 5)
   165→   165→}

   311→   311→    } catch {
