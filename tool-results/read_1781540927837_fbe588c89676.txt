   117→// ─── Smart Token Saver: Local, free, instant tagging ──────────────────
   118→function getLocalTags(text: string): string[] {
   119→  const lower = text.toLowerCase()
   120→  const tags: string[] = []
   121→
   122→  // URL detection → 'link'
   123→  if (/https?:\/\//.test(lower)) {
   124→    tags.push('link')
   125→  }
   126→
   127→  // Task words → 'task'
   128→  if (/\b(todo|remind|need to|must|buy)\b/.test(lower)) {
   129→    tags.push('task')
   130→  }
   131→
   132→  // Person/personal words → 'personal'
   133→  if (/\b(i |me |with |call|email)\b/.test(lower)) {
   134→    tags.push('personal')
   135→  }
   136→
   137→  // Question/idea words → 'idea'
   138→  if (/\b(what|how|why|who)\b/.test(lower)) {
   139→    tags.push('idea')
   140→  }
   141→
   142→  return tags.slice(0, 5)
   143→}

   145→// ─── Extended keyword-based auto-tagging (richer, for AI fallback) ───
   146→function autoGenerateTags(content: string, title: string): string[] {
   147→  const text = `${title} ${content}`.toLowerCase()
   148→  const tagMap: Record<string, string[]> = {
   149→    'work': ['meeting', 'project', 'deadline', 'client', 'office', 'team', 'q1', 'q2', 'q3', 'q4', 'quarterly', 'strategy'],
   150→    'personal': ['routine', 'morning', 'exercise', 'meditation', 'journal', 'habit'],
   151→    'travel': ['trip', 'itinerary', 'flight', 'hotel', 'visit', 'tokyo', 'paris', 'destination'],
   152→    'learning': ['learn', 'study', 'course', 'tutorial', 'book', 'read', 'article'],
   153→    'code': ['code', 'programming', 'react', 'javascript', 'typescript', 'api', 'bug', 'feature', 'css', 'html', 'framework'],
   154→    'design': ['design', 'ui', 'ux', 'layout', 'color', 'font', 'figma', 'wireframe'],
   155→    'ai': ['ai', 'machine learning', 'neural', 'model', 'gpt', 'gemini', 'llm', 'chatbot'],
   156→    'recipe': ['recipe', 'cook', 'bake', 'ingredient', 'food', 'meal', 'breakfast', 'dinner'],
   157→    'idea': ['idea', 'concept', 'brainstorm', 'innovative', 'startup', 'prototype'],
   158→    'finance': ['budget', 'expense', 'invest', 'savings', 'money', 'cost'],
   159→  }
   160→
   161→  const tags: string[] = []
   162→  for (const [tag, keywords] of Object.entries(tagMap)) {
   163→    if (keywords.some(kw => text.includes(kw))) {
   164→      tags.push(tag)
   165→    }
   166→  }
   167→  return tags.slice(0, 5)
   168→}

   665→  // ── Background link content reading (non-blocking) ──────────────────
   666→  backgroundReadLink: async (memoryId: string, url: string) => {

   669→        method: 'POST',
   670→        headers: { 'Content-Type': 'application/json' },

   674→      if (!res.ok) return
   675→

   677→      if (!content?.trim()) return
   678→
   679→      const state = get()
   680→
   681→      // Build enriched content: keep original URL + append extracted content
   682→      const enrichedContent = `${url}\n\n---\n\n${content}`.slice(0, 3000)
   683→
   684→      // Update in Supabase if authenticated + ready
   685→      if (state.isAuthenticated && state.supabaseReady) {
   686→        try {
   687→          const supabase = await getSupabaseBrowser()
   688→          const { data: { session } } = await supabase.auth.getSession()
   689→          if (session) {
   690→            const updateData: Record<string, string> = { content: enrichedContent }
   691→            if (title?.trim()) updateData.title = title
   692→            if (description?.trim()) updateData.summary = description
   693→            await supabase
   694→              .from('memories')
   695→              .update(updateData)
   696→              .eq('id', memoryId)
   697→          }
   698→        } catch {
   699→          // Silently fail
   700→        }
   701→      } else {
   702→        // Update via Prisma API
   703→        try {
   704→          const updateData: Record<string, unknown> = { content: enrichedContent }
   705→          if (title?.trim()) updateData.title = title
   706→          if (description?.trim()) updateData.summary = description
   707→          await fetch(`/api/memories/${memoryId}`, {
   708→            method: 'PATCH',
   709→            headers: { 'Content-Type': 'application/json' },
   710→            body: JSON.stringify(updateData),
   711→          })
   712→        } catch {
   713→          // Silently fail
   714→        }
   715→      }
   716→
   717→      // Update local state
   718→      const localUpdate: Partial<Memory> = { content: enrichedContent }
   719→      if (title?.trim()) localUpdate.title = title
   720→      if (description?.trim()) localUpdate.summary = description
   721→      get().updateMemory(memoryId, localUpdate)
   722→
   723→      // After reading the link, also tag and summarize it with AI
   724→      get().backgroundAutoTag(memoryId, enrichedContent, title || '')
   725→      get().backgroundAutoSummary(memoryId, enrichedContent)

   729→    }

   746→  // ── Save Memory (Supabase-aware, optimistic UI, token-saving) ────────
   747→  saveMemory: async (data) => {
   748→    const state = get()
   749→    const { type, title, content, sourceUrl, imageUrl, tags, collectionIds } = data
   750→
   751→    // ── STEP 1: Smart Token Saver — local, free, instant tags ──────────
   752→    const fullText = `${title || ''} ${content || ''}`.trim()
   753→    const localTags = getLocalTags(fullText)
   754→    const wordCount = fullText.split(/\s+/).filter(Boolean).length
   755→
   756→    // If <20 words AND local tags found: use only local tags, skip AI
   757→    const isShortNote = wordCount < 20 && localTags.length > 0
   758→
   759→    // Merge tags: caller tags > local tags > extended auto tags
   760→    let finalTags = tags
   761→    if (!tags || tags.length === 0) {
   762→      if (localTags.length > 0) {
   763→        finalTags = localTags
   764→      } else if (content?.trim()) {
   765→        const autoTags = autoGenerateTags(content, title || '')
   766→        if (autoTags.length > 0) finalTags = autoTags
   767→      }
   768→    }
   769→
   770→    // ── Authenticated + Supabase ready: save to Supabase directly ──
   771→    if (state.isAuthenticated && state.supabaseReady) {
   772→      try {
   773→        const supabase = await getSupabaseBrowser()
   774→        const { data: { session } } = await supabase.auth.getSession()
   775→        if (!session) throw new Error('No session')
   776→
   777→        const { data: memoryRow, error } = await supabase
   778→          .from('memories')
   779→          .insert({
   780→            user_id: session.user.id,
   781→            type: type || 'text',
   782→            title: title || '',
   783→            content: content || '',
   784→            source_url: sourceUrl || null,
   785→            image_url: imageUrl || null,
   786→            tags: finalTags ? (Array.isArray(finalTags) ? finalTags.join(',') : finalTags) : '',
   787→          })
   788→          .select('*, memory_collections(collection_id, collections(id, name, color, icon))')
   789→          .single()
   790→
   791→        if (error) throw error
   792→
   793→        const memory = mapSupabaseMemory(memoryRow as SupabaseMemoryRow)
   794→
   795→        // If collectionIds provided, create junction rows
   796→        if (collectionIds?.length) {
   797→          await supabase.from('memory_collections').insert(
   798→            collectionIds.map((cid: string) => ({
   799→              memory_id: memory.id,
   800→              collection_id: cid,
   801→            }))
   802→          )
   803→        }
   804→
   805→        // Optimistic UI: add to store instantly
   806→        get().addMemory(memory)
   807→
   808→        // ── STEP 2: Deferred AI tagging — ONLY if not a short note with local tags ──
   809→        if (content?.trim() && !isShortNote) {
   810→          get().backgroundAutoTag(memory.id, content, title || '')
   811→        }
   812→
   813→        // ── STEP 3: AI summary generation (background, non-blocking) ──
   814→        if (content?.trim() && !isShortNote) {
   815→          get().backgroundAutoSummary(memory.id, content)
   816→        }
   817→
   818→        // ── STEP 4: If this is a link, read the URL content (background, non-blocking) ──
   819→        if (sourceUrl?.trim()) {
   820→          get().backgroundReadLink(memory.id, sourceUrl)
   821→        }
   822→
   823→        // ── STEP 5: Generate embedding for semantic search (background, non-blocking) ──
   824→        if (content?.trim()) {
   825→          get().backgroundGenerateEmbedding(memory.id, content)
   826→        }
   827→
   828→        return memory
   829→      } catch {
   830→        // Fall through to Prisma API
   831→      }
   832→    }
   833→
   834→    // ── Fallback: save via Prisma API ──
   835→    try {
   836→      const res = await fetch('/api/memories', {
   837→        method: 'POST',
   838→        headers: { 'Content-Type': 'application/json' },
   839→        body: JSON.stringify({
   840→          type,
   841→          title,
   842→          content,
   843→          sourceUrl,
   844→          imageUrl,
   845→          tags: finalTags,
   846→          collectionIds,
   847→        }),
   848→      })
   849→      if (res.ok) {
   850→        const memory: Memory = await res.json()
   851→        get().addMemory(memory)
   852→
   853→        // ── STEP 2: Deferred AI tagging — ONLY if not a short note with local tags ──
   854→        if (content?.trim() && !isShortNote) {
   855→          get().backgroundAutoTag(memory.id, content, title || '')
   856→        }
   857→
   858→        // ── STEP 3: AI summary generation (background, non-blocking) ──
   859→        if (content?.trim() && !isShortNote) {
   860→          get().backgroundAutoSummary(memory.id, content)
   861→        }
   862→
   863→        // ── STEP 4: If this is a link, read the URL content (background, non-blocking) ──
   864→        if (sourceUrl?.trim()) {
   865→          get().backgroundReadLink(memory.id, sourceUrl)
   866→        }
