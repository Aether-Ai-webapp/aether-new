    45→// ─── Transcribe audio via z-ai-web-dev-sdk ASR ─────────────────────
    46→async function transcribeAudio(audioFile: File): Promise<string> {
    47→  try {
    48→    const { createTranscription } = await import('@/lib/aether-asr')
    49→    const transcript = await createTranscription(audioFile)
    50→    if (transcript?.trim()) return transcript.trim()
    51→  } catch (err) {
    52→    console.error('[capture] ASR transcription failed:', err instanceof Error ? err.message : 'Unknown error')
    53→  }
    54→
    55→  // Fallback: try Groq Whisper via REST API
    56→  try {
    57→    const groqKey = process.env.NEXT_PUBLIC_GROQ_API_KEY
    58→    if (!groqKey || groqKey === 'placeholder_groq_key') return ''
    59→
    60→    const arrayBuffer = await audioFile.arrayBuffer()
    61→    const blob = new Blob([arrayBuffer], { type: audioFile.type || 'audio/webm' })
    62→    const formData = new FormData()
    63→    formData.append('file', blob, audioFile.name || 'recording.webm')
    64→    formData.append('model', 'whisper-large-v3-turbo')
    65→    formData.append('response_format', 'json')
    66→
    67→    const response = await fetch('https://api.groq.com/openai/v1/audio/transcriptions', {
    68→      method: 'POST',
    69→      headers: { Authorization: `Bearer ${groqKey}` },
    70→      body: formData,
    71→    })
    72→
    73→    if (response.ok) {
    74→      const data = await response.json()
    75→      if (data.text?.trim()) return data.text.trim()
    76→    }
    77→  } catch (err) {
    78→    console.error('[capture] Groq Whisper fallback failed:', err instanceof Error ? err.message : 'Unknown error')
    79→  }
    80→
    81→  return ''
    82→}

   225→async function matchToExistingCollections(memoryId: string, memoryTags: string[]): Promise<void> {
   226→  try {
   227→    const db = await getPrisma()
   228→
   229→    // Get all collections
   230→    const collections = await db.collection.findMany()
   231→
   232→    for (const collection of collections) {
   233→      const collNameLower = collection.name.toLowerCase()
   234→
   235→      // Check if any memory tag matches or relates to the collection name
   236→      const tagMatches = memoryTags.some(
   237→        (tag) =>
   238→          collNameLower.includes(tag) ||
   239→          tag.includes(collNameLower) ||
   240→          levenshteinSimilarity(collNameLower, tag) > 0.6
   241→      )
   242→
   243→      if (tagMatches) {
   244→        // Check if the junction already exists
   245→        const existing = await db.memoryCollection.findFirst({
   246→          where: {
   247→            memoryId,
   248→            collectionId: collection.id,
   249→          },
   250→        })
   251→
   252→        if (!existing) {
   253→          await db.memoryCollection.create({
   254→            data: {
   255→              memoryId,
   256→              collectionId: collection.id,
   257→            },
   258→          })
   259→        }
   260→      }
   261→    }
   262→
   263→    // Also try Supabase path

   265→      const supabase = await getSupabaseServer()
   266→      const { data: { user } } = await supabase.auth.getUser()
   267→      if (user) {
   268→        const { data: supabaseCollections } = await supabase
   269→          .from('collections')
   270→          .select('id, name')
   271→          .eq('user_id', user.id)
   272→
   273→        if (supabaseCollections) {
   274→          for (const col of supabaseCollections) {
   275→            const colNameLower = col.name.toLowerCase()
   276→            const tagMatches = memoryTags.some(
   277→              (tag) =>
   278→                colNameLower.includes(tag) ||
   279→                tag.includes(colNameLower) ||
   280→                levenshteinSimilarity(colNameLower, tag) > 0.6
   281→            )

   283→              await supabase.from('memory_collections').upsert({
   284→                memory_id: memoryId,
   285→                collection_id: col.id,
   286→              }, { onConflict: 'memory_id,collection_id' })

   299→async function checkAutoCollectionRule(baseUrl: string): Promise<void> {
   300→  try {
   301→    const db = await getPrisma()
   302→
   303→    // Find memories that have NO collection assignments
   304→    const uncollectedMemories = await db.memory.findMany({
   305→      where: {
   306→        collections: { none: {} },
   307→        tags: { not: '' },
   308→      },
   309→      select: { id: true, tags: true },
   310→    })
   311→
   312→    if (uncollectedMemories.length < 10) return
   313→
   314→    // Build tag frequency map
   315→    const tagFrequency: Record<string, string[]> = {} // tag -> array of memory IDs
   316→    for (const mem of uncollectedMemories) {
   317→      const memTags = mem.tags.split(',').map((t) => t.trim().toLowerCase()).filter(Boolean)
   318→      for (const tag of memTags) {
   319→        if (!tagFrequency[tag]) tagFrequency[tag] = []
   320→        tagFrequency[tag].push(mem.id)
   321→      }
   322→    }
   323→
   324→    // Find tag clusters that hit 10+ memories
   325→    const clusterColors = ['#6D597A', '#B56576', '#355070', '#EA526F', '#23B5D3', '#7B2D8E', '#F18F01', '#C73E1D']
   326→    let colorIdx = await db.collection.count()
   327→
   328→    for (const [tag, memoryIds] of Object.entries(tagFrequency)) {
   329→      if (memoryIds.length >= 10) {
   330→        // Check if a collection for this tag already exists
   331→        const existingCollection = await db.collection.findFirst({
   332→          where: { name: { equals: tag.charAt(0).toUpperCase() + tag.slice(1), mode: 'insensitive' } },
   333→        })
   334→
   335→        if (!existingCollection) {
   336→          // Create a new collection with an intelligent name
   337→          const collectionName = tag.charAt(0).toUpperCase() + tag.slice(1)
   338→          const color = clusterColors[colorIdx % clusterColors.length]
   339→          colorIdx++
   340→
   341→          const newCollection = await db.collection.create({
   342→            data: {
   343→              name: collectionName,
   344→              icon: getTagIcon(tag),
   345→              color,
   346→            },
   347→          })
   348→
   349→          // Assign the clustered memories to this new collection
   350→          for (const memId of memoryIds) {
   351→            await db.memoryCollection.upsert({
   352→              where: {
   353→                memoryId_collectionId: { memoryId: memId, collectionId: newCollection.id },
   354→              },
   355→              create: {
   356→                memoryId: memId,
   357→                collectionId: newCollection.id,
   358→              },
   359→              update: {},
   360→            })
   361→          }
   362→
   363→          console.log(`[capture] Auto-created collection "${collectionName}" with ${memoryIds.length} memories`)
   364→
   365→          // Also create in Supabase if available
   366→          try {
   367→            const supabase = await getSupabaseServer()
   368→            const { data: { user } } = await supabase.auth.getUser()
   369→            if (user) {
   370→              const { data: supaCol } = await supabase
   371→                .from('collections')
   372→                .insert({
   373→                  user_id: user.id,
   374→                  name: collectionName,
   375→                  icon: getTagIcon(tag),
   376→                  color,
   377→                })
   378→                .select('id')
   379→                .single()
   380→
   381→              if (supaCol) {
   382→                for (const memId of memoryIds.slice(0, 50)) {
   383→                  await supabase.from('memory_collections').upsert({
   384→                    memory_id: memId,
   385→                    collection_id: supaCol.id,
   386→                  }, { onConflict: 'memory_id,collection_id' })
   387→                }
   388→              }
   389→            }
   390→          } catch {
   391→            // Supabase path optional
   392→          }
   393→        }
   394→      }
   395→    }
   396→  } catch (err) {
   397→    console.error('[capture] Auto-collection rule error:', err instanceof Error ? err.message : 'Unknown error')

   412→// ─── Levenshtein similarity for fuzzy tag matching ─────────────────
   413→function levenshteinSimilarity(a: string, b: string): number {
   414→  if (a === b) return 1
   415→  if (!a.length || !b.length) return 0
   416→
   417→  const matrix: number[][] = []
   418→  for (let i = 0; i <= b.length; i++) matrix[i] = [i]
   419→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j
   420→
   421→  for (let i = 1; i <= b.length; i++) {
   422→    for (let j = 1; j <= a.length; j++) {
   423→      const cost = b[i - 1] === a[j - 1] ? 0 : 1
   424→      matrix[i][j] = Math.min(
   425→        matrix[i - 1][j] + 1,
   426→        matrix[i][j - 1] + 1,
   427→        matrix[i - 1][j - 1] + cost
   428→      )
   429→    }
   430→  }
   431→
   432→  const maxLen = Math.max(a.length, b.length)
   433→  return 1 - matrix[b.length][a.length] / maxLen
   434→}

   527→        try {
   528→          const arrayBuffer = await imageFile.arrayBuffer()
   529→          const base64 = Buffer.from(arrayBuffer).toString('base64')
   530→          const mimeType = imageFile.type || 'image/png'
   531→          imagePreviewUrl = `data:${mimeType};base64,${base64}`
   532→        } catch {
   533→          imagePreviewUrl = null
   534→        }
   535→      }
   536→    }
   537→
   538→    // 3c: Determine the content type
   539→    let memoryType = 'text'
   540→    if (typeOverride && ['text', 'voice', 'link', 'image'].includes(typeOverride)) {
   541→      memoryType = typeOverride
   542→    } else if (hasImage) {
   543→      memoryType = 'image'
   544→    } else if (hasAudio && !hasText && !hasUrl) {
   545→      memoryType = 'voice'
   546→    } else if (hasUrl || (hasText && /https?:\/\//.test(textInput.toLowerCase()))) {
   547→      memoryType = 'link'
   548→    }
   549→
   550→    // 3d: Compose the raw content string
   551→    let rawContent = ''
   552→    if (audioTranscript.trim()) {
   553→      rawContent = audioTranscript.trim()
   554→      if (hasText && textInput.trim()) {
   555→        rawContent = `${textInput.trim()}\n\n${audioTranscript.trim()}`
   556→      }
   557→    } else if (hasText) {
   558→      rawContent = textInput.trim()
   559→    }
   560→
   561→    const sourceUrl = hasUrl ? urlInput.trim() : memoryType === 'link' ? textInput.trim() : null
   562→
   563→    if (hasImage && !rawContent.trim()) {
   564→      rawContent = imageFile ? `Image: ${imageFile.name}` : 'Image'
   565→    }
   566→
   567→    // Compose the title
   568→    let title = ''
   569→    if (rawContent.length > 80) {
   570→      title = rawContent.slice(0, 80) + '...'
   571→    } else {
   572→      title = rawContent
   573→    }
   574→    if (hasImage && !textInput.trim()) {
   575→      title = imageFile ? `Image: ${imageFile.name}` : 'Image'
   576→    }
   577→
   578→    // ── STEP 4: COGNITIVE SYNTHESIS — Gemini Flash ─────────────────
   579→    const textForSynthesis = rawContent || (hasUrl ? urlInput : '')
   580→    let cleanedSummary: string | null = null
   581→    let deepInsight: string | null = null
   582→    let aiTags: string[] = []
   583→
   584→    // Start with keyword tags (instant, free)
   585→    const keywordTags = autoGenerateTags(rawContent, title)
   586→
   587→    if (textForSynthesis.trim()) {
   588→      const cognitive = await generateCognitiveSynthesis(textForSynthesis)
   589→      if (cognitive.summary) cleanedSummary = cognitive.summary
   590→      if (cognitive.deepInsight) deepInsight = cognitive.deepInsight
   591→      if (cognitive.tags.length > 0) aiTags = cognitive.tags
   592→    }
   593→
   594→    // Merge: AI tags take priority, keyword tags fill gaps
   595→    const finalTags = aiTags.length > 0
   596→      ? [...new Set([...aiTags, ...keywordTags])].slice(0, 5)
   597→      : keywordTags
   598→
   599→    // ── STEP 5: Database ingestion ───────────────────────────────────
   600→    let memory: Record<string, unknown>
   601→
   602→    // 5a: Try Supabase insertion
   603→    if (useSupabase && supabaseSession) {
   604→      try {
   605→        const supabase = await getSupabaseServer()
   606→        const insertData: Record<string, unknown> = {
   607→          user_id: supabaseSession.user.id,
   608→          type: memoryType,
   609→          title,
   610→          content: rawContent,
   611→          source_url: sourceUrl || null,
   612→          tags: finalTags.join(','),
   613→        }
   614→
   615→        if (cleanedSummary) insertData.summary = cleanedSummary
   616→        if (deepInsight) insertData.deep_insight = deepInsight
   617→        if (imageUrl) insertData.image_url = imageUrl
   618→        if (imagePreviewUrl && !imageUrl) insertData.image_preview = imagePreviewUrl
   619→
   620→        const { data: memoryRow, error: insertError } = await supabase
   621→          .from('memories')
   622→          .insert(insertData)
   623→          .select('*, memory_collections(collection_id, collections(id, name, color, icon))')
   624→          .single()
   625→
   626→        if (insertError) throw insertError
   627→
   628→        const row = memoryRow as Record<string, unknown>
   629→        const collectionsFromRow = (
   630→          (row.memory_collections as Record<string, Record<string, unknown>>[]) || []
   631→        ).map((mc) => ({
   632→          id: (mc.collections as Record<string, unknown>)?.id as string,
   633→          name: (mc.collections as Record<string, unknown>)?.name as string,
   634→          color: (mc.collections as Record<string, unknown>)?.color as string,
   635→          icon: (mc.collections as Record<string, unknown>)?.icon as string,
   636→        }))
   637→
   638→        memory = buildMemoryResponse(row, collectionsFromRow)
   639→
   640→        // ── Background: embedding + link reading + auto-collections ────
   641→        if (rawContent.trim()) {
   642→          fetch(`${baseUrl}/api/generate-embedding`, {
   643→            method: 'POST',
   644→            headers: { 'Content-Type': 'application/json' },
   645→            body: JSON.stringify({ memoryId: memory.id, content: rawContent.slice(0, 2000) }),
   646→          }).catch(() => {})
   647→        }
   648→
   649→        if (sourceUrl?.trim()) {
   650→          fetch(`${baseUrl}/api/ai/read-link`, {
   651→            method: 'POST',
   652→            headers: { 'Content-Type': 'application/json' },
   653→            body: JSON.stringify({ url: sourceUrl }),
   654→          }).catch(() => {})
   655→        }
   656→
   657→        // Autonomous collections (fire-and-forget)
   658→        runAutonomousCollections(memory.id as string, finalTags, baseUrl).catch(() => {})
   659→
   660→        return NextResponse.json({ success: true, memory })
   661→      } catch (err) {
   662→        console.error('[capture] Supabase insertion failed, falling back to Prisma:', err instanceof Error ? err.message : 'Unknown error')
   663→      }
   664→    }
   665→
   666→    // 5b: Fallback — Prisma SQLite insertion
   667→    try {
   668→      const db = await getPrisma()
   669→
   670→      const prismaMemory = await db.memory.create({
   671→        data: {
   672→          type: memoryType,
   673→          title,
   674→          content: rawContent,
   675→          summary: cleanedSummary,
   676→          deepInsight,
   677→          sourceUrl: sourceUrl || null,
   678→          tags: finalTags.join(','),
   679→          imageUrl: imageUrl || null,
   680→          imagePreview: imagePreviewUrl && !imageUrl ? imagePreviewUrl : null,
   681→        },
   682→        include: {
   683→          collections: {
   684→            include: {
   685→              collection: {
   686→                select: { id: true, name: true, color: true, icon: true },
   687→              },
   688→            },
   689→          },
   690→        },
   691→      })
   692→
   693→      memory = buildMemoryResponse(
   694→        {
   695→          id: prismaMemory.id,
   696→          type: prismaMemory.type,
   697→          title: prismaMemory.title,
   698→          content: prismaMemory.content,
   699→          summary: prismaMemory.summary,
   700→          deepInsight: prismaMemory.deepInsight,
   701→          tags: prismaMemory.tags,
   702→          sourceUrl: prismaMemory.sourceUrl,
   703→          fileUrl: prismaMemory.fileUrl,
   704→          imagePreview: prismaMemory.imagePreview,
   705→          imageUrl: prismaMemory.imageUrl,
   706→          isFavorite: prismaMemory.isFavorite,
   707→          created_at: prismaMemory.createdAt.toISOString(),
   708→          updated_at: prismaMemory.updatedAt.toISOString(),
   709→        },
   710→        prismaMemory.collections.map((mc) => ({
   711→          id: mc.collection.id,
   712→          name: mc.collection.name,
   713→          color: mc.collection.color,
   714→          icon: mc.collection.icon,
   715→        }))
   716→      )
   717→
   718→      // ── Background: embedding + link reading + auto-collections ────
   719→      if (rawContent.trim()) {
   720→        fetch(`${baseUrl}/api/generate-embedding`, {
   721→          method: 'POST',
   722→          headers: { 'Content-Type': 'application/json' },
   723→          body: JSON.stringify({ memoryId: prismaMemory.id, content: rawContent.slice(0, 2000) }),
   724→        }).catch(() => {})
   725→      }
   726→
   727→      if (sourceUrl?.trim()) {
   728→        fetch(`${baseUrl}/api/ai/read-link`, {
   729→          method: 'POST',
   730→          headers: { 'Content-Type': 'application/json' },
   731→          body: JSON.stringify({ url: sourceUrl }),
   732→        }).catch(() => {})
   733→      }
   734→
   735→      // Autonomous collections (fire-and-forget)
   736→      runAutonomousCollections(prismaMemory.id, finalTags, baseUrl).catch(() => {})
   737→
   738→      return NextResponse.json({ success: true, memory })
   739→    } catch (err) {
   740→      console.error('[capture] Prisma insertion failed:', err instanceof Error ? err.message : 'Unknown error')
   741→      return NextResponse.json(
   742→        { success: false, error: 'Failed to save memory to database' },
   743→        { status: 500 }
   744→      )
   745→    }
   746→  } catch (error) {
   747→    console.error('[capture] Unhandled error:', error instanceof Error ? error.message : 'Unknown error')
   748→    return NextResponse.json(
   749→      { success: false, error: 'Internal server error during capture' },
   750→      { status: 500 }
   751→    )
   752→  }
   753→}
