    74→function mapSupabaseMemory(row: SupabaseMemoryRow): Memory {
    75→  return {
    76→    id: row.id,
    77→    type: (row.type as MemoryType) || 'text',
    78→    title: row.title || '',
    79→    content: row.content || '',
    80→    summary: row.summary,
    81→    tags: row.tags ? row.tags.split(',').filter(Boolean) : [],
    82→    sourceUrl: row.source_url,
    83→    fileUrl: row.file_url,
    84→    imagePreview: row.image_preview,
    85→    isFavorite: row.is_favorite || false,
    86→    createdAt: row.created_at,
    87→    updatedAt: row.updated_at,
    88→    collections: row.memory_collections?.map((mc) => ({
    89→      id: mc.collections.id,
    90→      name: mc.collections.name,
    91→      color: mc.collections.color,
    92→      icon: mc.collections.icon,

   108→// ─── Smart Token Saver: Local, free, instant tagging ──────────────────
   109→function getLocalTags(text: string): string[] {
   110→  const lower = text.toLowerCase()
   111→  const tags: string[] = []
   112→
   113→  // URL detection → 'link'
   114→  if (/https?:\/\//.test(lower)) {
   115→    tags.push('link')
   116→  }
   117→
   118→  // Task words → 'task'
   119→  if (/\b(todo|remind|need to|must|buy)\b/.test(lower)) {
   120→    tags.push('task')
   121→  }
   122→
   123→  // Person/personal words → 'personal'
   124→  if (/\b(i |me |with |call|email)\b/.test(lower)) {
   125→    tags.push('personal')
   126→  }
   127→
   128→  // Question/idea words → 'idea'
   129→  if (/\b(what|how|why|who)\b/.test(lower)) {
   130→    tags.push('idea')
   131→  }
   132→
   133→  return tags.slice(0, 5)
   134→}

   136→// ─── Extended keyword-based auto-tagging (richer, for AI fallback) ───
   137→function autoGenerateTags(content: string, title: string): string[] {
   138→  const text = `${title} ${content}`.toLowerCase()
   139→  const tagMap: Record<string, string[]> = {
   140→    'work': ['meeting', 'project', 'deadline', 'client', 'office', 'team', 'q1', 'q2', 'q3', 'q4', 'quarterly', 'strategy'],
   141→    'personal': ['routine', 'morning', 'exercise', 'meditation', 'journal', 'habit'],
   142→    'travel': ['trip', 'itinerary', 'flight', 'hotel', 'visit', 'tokyo', 'paris', 'destination'],
   143→    'learning': ['learn', 'study', 'course', 'tutorial', 'book', 'read', 'article'],
   144→    'code': ['code', 'programming', 'react', 'javascript', 'typescript', 'api', 'bug', 'feature', 'css', 'html', 'framework'],
   145→    'design': ['design', 'ui', 'ux', 'layout', 'color', 'font', 'figma', 'wireframe'],
   146→    'ai': ['ai', 'machine learning', 'neural', 'model', 'gpt', 'gemini', 'llm', 'chatbot'],
   147→    'recipe': ['recipe', 'cook', 'bake', 'ingredient', 'food', 'meal', 'breakfast', 'dinner'],
   148→    'idea': ['idea', 'concept', 'brainstorm', 'innovative', 'startup', 'prototype'],
   149→    'finance': ['budget', 'expense', 'invest', 'savings', 'money', 'cost'],
   150→  }
   151→
   152→  const tags: string[] = []
   153→  for (const [tag, keywords] of Object.entries(tagMap)) {
   154→    if (keywords.some(kw => text.includes(kw))) {
   155→      tags.push(tag)
   156→    }
   157→  }
   158→  return tags.slice(0, 5)
   159→}

   556→  // ── Background AI auto-tagging (non-blocking) ──────────────────────
   557→  backgroundAutoTag: async (memoryId: string, content: string, title: string) => {
   558→    try {
   559→      const res = await fetch('/api/auto-tag', {
   560→        method: 'POST',
   561→        headers: { 'Content-Type': 'application/json' },
   562→        body: JSON.stringify({ content: `${title} ${content}`.slice(0, 1000) }),
   563→      })
   564→
   565→      if (!res.ok) return
   566→
   567→      const { tags: aiTags } = await res.json()
   568→      if (!aiTags || !aiTags.length) return
   569→
   570→      const state = get()
   571→
   572→      // Update in Supabase if authenticated + ready
   573→      if (state.isAuthenticated && state.supabaseReady) {
   574→        try {
   575→          const supabase = await getSupabaseBrowser()
   576→          const { data: { session } } = await supabase.auth.getSession()
   577→          if (session) {
   578→            await supabase
   579→              .from('memories')
   580→              .update({ tags: aiTags.join(',') })
   581→              .eq('id', memoryId)
   582→          }
   583→        } catch {
   584→          // Silently fail Supabase update — local state still updates
   585→        }
   586→      } else {
   587→        // Update via Prisma API
   588→        try {
   589→          await fetch(`/api/memories/${memoryId}`, {
   590→            method: 'PATCH',
   591→            headers: { 'Content-Type': 'application/json' },
   592→            body: JSON.stringify({ tags: aiTags }),
   593→          })
   594→        } catch {
   595→          // Silently fail
   596→        }
   597→      }
   598→
   599→      // Update the local state so tags appear immediately
   600→      get().updateMemory(memoryId, { tags: aiTags })
   601→    } catch {
   602→      // Non-blocking: never fail the save flow
   603→    }
   604→  },

   737→  // ── Save Memory (Supabase-aware, optimistic UI, token-saving) ────────
   738→  saveMemory: async (data) => {
   739→    const state = get()
   740→    const { type, title, content, sourceUrl, tags, collectionIds } = data
   741→
   742→    // ── STEP 1: Smart Token Saver — local, free, instant tags ──────────
   743→    const fullText = `${title || ''} ${content || ''}`.trim()
   744→    const localTags = getLocalTags(fullText)
   745→    const wordCount = fullText.split(/\s+/).filter(Boolean).length
   746→
   747→    // If <20 words AND local tags found: use only local tags, skip AI
   748→    const isShortNote = wordCount < 20 && localTags.length > 0
   749→
   750→    // Merge tags: caller tags > local tags > extended auto tags
   751→    let finalTags = tags
   752→    if (!tags || tags.length === 0) {
   753→      if (localTags.length > 0) {
   754→        finalTags = localTags
   755→      } else if (content?.trim()) {
   756→        const autoTags = autoGenerateTags(content, title || '')
   757→        if (autoTags.length > 0) finalTags = autoTags
   758→      }
   759→    }
   760→
   761→    // ── Authenticated + Supabase ready: save to Supabase directly ──
   762→    if (state.isAuthenticated && state.supabaseReady) {
   763→      try {
   764→        const supabase = await getSupabaseBrowser()
   765→        const { data: { session } } = await supabase.auth.getSession()
   766→        if (!session) throw new Error('No session')
   767→
   768→        const { data: memoryRow, error } = await supabase
   769→          .from('memories')
   770→          .insert({
   771→            user_id: session.user.id,
   772→            type: type || 'text',
   773→            title: title || '',
   774→            content: content || '',
   775→            source_url: sourceUrl || null,
   776→            tags: finalTags ? (Array.isArray(finalTags) ? finalTags.join(',') : finalTags) : '',
   777→          })
   778→          .select('*, memory_collections(collection_id, collections(id, name, color, icon))')
   779→          .single()
   780→
   781→        if (error) throw error
   782→
   783→        const memory = mapSupabaseMemory(memoryRow as SupabaseMemoryRow)
   784→
   785→        // If collectionIds provided, create junction rows
   786→        if (collectionIds?.length) {
   787→          await supabase.from('memory_collections').insert(
   788→            collectionIds.map((cid: string) => ({
   789→              memory_id: memory.id,
   790→              collection_id: cid,
   791→            }))
   792→          )
   793→        }
   794→
   795→        // Optimistic UI: add to store instantly
   796→        get().addMemory(memory)
   797→
   798→        // ── STEP 2: Deferred AI tagging — ONLY if not a short note with local tags ──
   799→        if (content?.trim() && !isShortNote) {
   800→          get().backgroundAutoTag(memory.id, content, title || '')
   801→        }
   802→
   803→        // ── STEP 3: AI summary generation (background, non-blocking) ──
   804→        if (content?.trim() && !isShortNote) {
   805→          get().backgroundAutoSummary(memory.id, content)
   806→        }
   807→
   808→        // ── STEP 4: If this is a link, read the URL content (background, non-blocking) ──
   809→        if (sourceUrl?.trim()) {
   810→          get().backgroundReadLink(memory.id, sourceUrl)
   811→        }
   812→
   813→        // ── STEP 5: Generate embedding for semantic search (background, non-blocking) ──
   814→        if (content?.trim()) {
   815→          get().backgroundGenerateEmbedding(memory.id, content)
   816→        }
   817→
   818→        return memory
   819→      } catch {
   820→        // Fall through to Prisma API
   821→      }
   822→    }
   823→
   824→    // ── Fallback: save via Prisma API ──
   825→    try {
   826→      const res = await fetch('/api/memories', {
   827→        method: 'POST',
   828→        headers: { 'Content-Type': 'application/json' },
   829→        body: JSON.stringify({
   830→          type,
   831→          title,
   832→          content,
   833→          sourceUrl,
   834→          tags: finalTags,
   835→          collectionIds,
   836→        }),
   837→      })
   838→      if (res.ok) {
   839→        const memory: Memory = await res.json()
   840→        get().addMemory(memory)
   841→
   842→        // ── STEP 2: Deferred AI tagging — ONLY if not a short note with local tags ──
   843→        if (content?.trim() && !isShortNote) {
   844→          get().backgroundAutoTag(memory.id, content, title || '')
   845→        }
   846→
   847→        // ── STEP 3: AI summary generation (background, non-blocking) ──
   848→        if (content?.trim() && !isShortNote) {
   849→          get().backgroundAutoSummary(memory.id, content)
   850→        }
   851→
   852→        // ── STEP 4: If this is a link, read the URL content (background, non-blocking) ──
   853→        if (sourceUrl?.trim()) {
   854→          get().backgroundReadLink(memory.id, sourceUrl)
   855→        }
