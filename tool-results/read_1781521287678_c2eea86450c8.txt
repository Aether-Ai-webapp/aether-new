   225→   225→async function matchToExistingCollections(memoryId: string, memoryTags: string[]): Promise<void> {

   227→   227→    const db = await getPrisma()
   228→   228→
   229→   229→    // Get all collections

   231→   231→

   234→   234→
   235→   235→      // Check if any memory tag matches or relates to the collection name
   236→   236→      const tagMatches = memoryTags.some(
   237→   237→        (tag) =>
   238→   238→          collNameLower.includes(tag) ||
   239→   239→          tag.includes(collNameLower) ||
   240→   240→          levenshteinSimilarity(collNameLower, tag) > 0.6
   241→   241→      )
   242→   242→
   243→   243→      if (tagMatches) {
   244→   244→        // Check if the junction already exists

   247→   247→            memoryId,

   255→   255→              memoryId,

   257→   257→            },
   258→   258→          })
   259→   259→        }
   260→   260→      }

   412→   412→// ─── Levenshtein similarity for fuzzy tag matching ─────────────────
   413→   413→function levenshteinSimilarity(a: string, b: string): number {
   414→   414→  if (a === b) return 1
   415→   415→  if (!a.length || !b.length) return 0
   416→   416→
   417→   417→  const matrix: number[][] = []
   418→   418→  for (let i = 0; i <= b.length; i++) matrix[i] = [i]
   419→   419→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j
   420→   420→
   421→   421→  for (let i = 1; i <= b.length; i++) {
   422→   422→    for (let j = 1; j <= a.length; j++) {
   423→   423→      const cost = b[i - 1] === a[j - 1] ? 0 : 1
   424→   424→      matrix[i][j] = Math.min(
   425→   425→        matrix[i - 1][j] + 1,
   426→   426→        matrix[i][j - 1] + 1,
   427→   427→        matrix[i - 1][j - 1] + cost
   428→   428→      )
   429→   429→    }
   430→   430→  }
   431→   431→
   432→   432→  const maxLen = Math.max(a.length, b.length)
   433→   433→  return 1 - matrix[b.length][a.length] / maxLen
   434→   434→}

   446→   446→    summary: (row.summary as string) || null,
   447→   447→    deepInsight: (row.deepInsight as string) || (row.deep_insight as string) || null,
   448→   448→    tags: row.tags ? (row.tags as string).split(',').filter(Boolean) : [],
   449→   449→    sourceUrl: row.source_url ?? row.sourceUrl ?? null,
   450→   450→    fileUrl: row.file_url ?? row.fileUrl ?? null,
   451→   451→    imagePreview: row.image_preview ?? row.imagePreview ?? null,
   452→   452→    imageUrl: row.image_url ?? row.imageUrl ?? null,
   453→   453→    recap: row.recap ?? null,
   454→   454→    isFavorite: (row.is_favorite ?? row.isFavorite ?? false) as boolean,
   455→   455→    createdAt: (row.created_at ?? row.createdAt ?? new Date().toISOString()) as string,
   456→   456→    updatedAt: (row.updated_at ?? row.updatedAt ?? new Date().toISOString()) as string,
   457→   457→    collections,
   458→   458→  }
   459→   459→}

   461→   461→// ═══════════════════════════════════════════════════════════════════════
   462→   462→// ─── POST /api/capture — THE UNIVERSAL CAPTURE ENGINE ────────────────
   463→   463→// ═══════════════════════════════════════════════════════════════════════
   464→   464→export async function POST(req: NextRequest) {
   465→   465→  try {
   466→   466→    // ── STEP 1: UNIVERSAL PAYLOAD SCANNER ─────────────────────────────
   467→   467→    // Parse the incoming FormData and classify each field by type.
   468→   468→    const formData = await req.formData()
   469→   469→
   470→   470→    const textInput = (formData.get('text') as string) || ''
   471→   471→    const urlInput = (formData.get('url') as string) || ''
   472→   472→    const typeOverride = (formData.get('type') as string) || ''
   473→   473→    const audioFile = formData.get('audio') as File | null
   474→   474→    const imageFile = formData.get('image') as File | null
   475→   475→
   476→   476→    const baseUrl = new URL(req.url).origin
   477→   477→
   478→   478→    // Classify: is it text? a link? an audio blob? an image file?
   479→   479→    const hasText = textInput.trim().length > 0
   480→   480→    const hasUrl = urlInput.trim().length > 0
   481→   481→    const hasAudio = audioFile !== null && audioFile.size > 0
   482→   482→    const hasImage = imageFile !== null && imageFile.size > 0
   483→   483→
   484→   484→    if (!hasText && !hasUrl && !hasAudio && !hasImage) {
   485→   485→      return NextResponse.json(
   486→   486→        { success: false, error: 'No content provided. Send text, audio, image, or url.' },
   487→   487→        { status: 400 }
   488→   488→      )
   489→   489→    }
   490→   490→
   491→   491→    // ── STEP 2: Determine the user (Supabase session or local) ─────
   492→   492→    let supabaseSession: { user: { id: string } } | null = null
   493→   493→    let useSupabase = false
   494→   494→
   495→   495→    try {
   496→   496→      const supabase = await getSupabaseServer()
   497→   497→      const { data: { user }, error: authError } = await supabase.auth.getUser()
   498→   498→      if (!authError && user) {
   499→   499→        supabaseSession = { user }
   500→   500→        const { error: tableCheck } = await supabase
   501→   501→          .from('memories')
   502→   502→          .select('id')
   503→   503→          .eq('user_id', user.id)
   504→   504→          .limit(1)
   505→   505→        useSupabase = !tableCheck
   506→   506→      }
   507→   507→    } catch {
   508→   508→      // No Supabase session
   509→   509→    }
   510→   510→
   511→   511→    // ── STEP 3: DYNAMIC INPUT DISPATCHING ────────────────────────────
   512→   512→
   513→   513→    // 3a: Transcribe audio if present
   514→   514→    let audioTranscript = ''
   515→   515→    if (hasAudio && audioFile) {
   516→   516→      audioTranscript = await transcribeAudio(audioFile)
   517→   517→    }
   518→   518→
   519→   519→    // 3b: Upload image if present
   520→   520→    let imageUrl: string | null = null
   521→   521→    let imagePreviewUrl: string | null = null
   522→   522→    if (hasImage && imageFile) {
   523→   523→      if (useSupabase && supabaseSession) {
   524→   524→        imageUrl = await uploadImageToStorage(imageFile, supabaseSession.user.id)
   525→   525→      }
   526→   526→      if (!imageUrl) {
   527→   527→        try {
   528→   528→          const arrayBuffer = await imageFile.arrayBuffer()
   529→   529→          const base64 = Buffer.from(arrayBuffer).toString('base64')
   530→   530→          const mimeType = imageFile.type || 'image/png'
   531→   531→          imagePreviewUrl = `data:${mimeType};base64,${base64}`
   532→   532→        } catch {
   533→   533→          imagePreviewUrl = null
   534→   534→        }
   535→   535→      }
   536→   536→    }
   537→   537→
   538→   538→    // 3c: Determine the content type
   539→   539→    let memoryType = 'text'
   540→   540→    if (typeOverride && ['text', 'voice', 'link', 'image'].includes(typeOverride)) {
   541→   541→      memoryType = typeOverride
   542→   542→    } else if (hasImage) {
   543→   543→      memoryType = 'image'
   544→   544→    } else if (hasAudio && !hasText && !hasUrl) {
   545→   545→      memoryType = 'voice'
   546→   546→    } else if (hasUrl || (hasText && /https?:\/\//.test(textInput.toLowerCase()))) {
   547→   547→      memoryType = 'link'
   548→   548→    }
   549→   549→
   550→   550→    // 3d: Compose the raw content string
   551→   551→    let rawContent = ''
   552→   552→    if (audioTranscript.trim()) {
   553→   553→      rawContent = audioTranscript.trim()
   554→   554→      if (hasText && textInput.trim()) {
   555→   555→        rawContent = `${textInput.trim()}\n\n${audioTranscript.trim()}`
   556→   556→      }
   557→   557→    } else if (hasText) {
   558→   558→      rawContent = textInput.trim()
   559→   559→    }
   560→   560→
   561→   561→    const sourceUrl = hasUrl ? urlInput.trim() : memoryType === 'link' ? textInput.trim() : null
   562→   562→
   563→   563→    if (hasImage && !rawContent.trim()) {
   564→   564→      rawContent = imageFile ? `Image: ${imageFile.name}` : 'Image'
   565→   565→    }
   566→   566→
   567→   567→    // Compose the title
   568→   568→    let title = ''
   569→   569→    if (rawContent.length > 80) {
   570→   570→      title = rawContent.slice(0, 80) + '...'
   571→   571→    } else {
   572→   572→      title = rawContent
   573→   573→    }
   574→   574→    if (hasImage && !textInput.trim()) {
   575→   575→      title = imageFile ? `Image: ${imageFile.name}` : 'Image'
   576→   576→    }
   577→   577→
   578→   578→    // ── STEP 4: COGNITIVE SYNTHESIS — Gemini Flash ─────────────────
   579→   579→    const textForSynthesis = rawContent || (hasUrl ? urlInput : '')
   580→   580→    let cleanedSummary: string | null = null
   581→   581→    let deepInsight: string | null = null
   582→   582→    let aiTags: string[] = []
   583→   583→
   584→   584→    // Start with keyword tags (instant, free)
   585→   585→    const keywordTags = autoGenerateTags(rawContent, title)
   586→   586→
   587→   587→    if (textForSynthesis.trim()) {
   588→   588→      const cognitive = await generateCognitiveSynthesis(textForSynthesis)
   589→   589→      if (cognitive.summary) cleanedSummary = cognitive.summary
   590→   590→      if (cognitive.deepInsight) deepInsight = cognitive.deepInsight
   591→   591→      if (cognitive.tags.length > 0) aiTags = cognitive.tags
   592→   592→    }
   593→   593→
   594→   594→    // Merge: AI tags take priority, keyword tags fill gaps
   595→   595→    const finalTags = aiTags.length > 0
   596→   596→      ? [...new Set([...aiTags, ...keywordTags])].slice(0, 5)
   597→   597→      : keywordTags
   598→   598→
   599→   599→    // ── STEP 5: Database ingestion ───────────────────────────────────
   600→   600→    let memory: Record<string, unknown>
   601→   601→
   602→   602→    // 5a: Try Supabase insertion
   603→   603→    if (useSupabase && supabaseSession) {
   604→   604→      try {
   605→   605→        const supabase = await getSupabaseServer()
   606→   606→        const insertData: Record<string, unknown> = {
   607→   607→          user_id: supabaseSession.user.id,
   608→   608→          type: memoryType,
   609→   609→          title,
   610→   610→          content: rawContent,
   611→   611→          source_url: sourceUrl || null,
   612→   612→          tags: finalTags.join(','),
   613→   613→        }
   614→   614→
   615→   615→        if (cleanedSummary) insertData.summary = cleanedSummary
   616→   616→        if (deepInsight) insertData.deep_insight = deepInsight
   617→   617→        if (imageUrl) insertData.image_url = imageUrl
   618→   618→        if (imagePreviewUrl && !imageUrl) insertData.image_preview = imagePreviewUrl
   619→   619→
   620→   620→        const { data: memoryRow, error: insertError } = await supabase
   621→   621→          .from('memories')
   622→   622→          .insert(insertData)
   623→   623→          .select('*, memory_collections(collection_id, collections(id, name, color, icon))')
   624→   624→          .single()
   625→   625→
   626→   626→        if (insertError) throw insertError
   627→   627→
   628→   628→        const row = memoryRow as Record<string, unknown>
   629→   629→        const collectionsFromRow = (
   630→   630→          (row.memory_collections as Record<string, Record<string, unknown>>[]) || []
   631→   631→        ).map((mc) => ({
   632→   632→          id: (mc.collections as Record<string, unknown>)?.id as string,
   633→   633→          name: (mc.collections as Record<string, unknown>)?.name as string,
   634→   634→          color: (mc.collections as Record<string, unknown>)?.color as string,
   635→   635→          icon: (mc.collections as Record<string, unknown>)?.icon as string,
   636→   636→        }))
   637→   637→
   638→   638→        memory = buildMemoryResponse(row, collectionsFromRow)
   639→   639→
   640→   640→        // ── Background: embedding + link reading + auto-collections ────
   641→   641→        if (rawContent.trim()) {
   642→   642→          fetch(`${baseUrl}/api/generate-embedding`, {
   643→   643→            method: 'POST',
   644→   644→            headers: { 'Content-Type': 'application/json' },
   645→   645→            body: JSON.stringify({ memoryId: memory.id, content: rawContent.slice(0, 2000) }),
   646→   646→          }).catch(() => {})
   647→   647→        }
   648→   648→
   649→   649→        if (sourceUrl?.trim()) {
   650→   650→          fetch(`${baseUrl}/api/ai/read-link`, {
   651→   651→            method: 'POST',
   652→   652→            headers: { 'Content-Type': 'application/json' },
   653→   653→            body: JSON.stringify({ url: sourceUrl }),
   654→   654→          }).catch(() => {})
   655→   655→        }
   656→   656→
   657→   657→        // Autonomous collections (fire-and-forget)
   658→   658→        runAutonomousCollections(memory.id as string, finalTags, baseUrl).catch(() => {})
   659→   659→
   660→   660→        return NextResponse.json({ success: true, memory })
   661→   661→      } catch (err) {
   662→   662→        console.error('[capture] Supabase insertion failed, falling back to Prisma:', err instanceof Error ? err.message : 'Unknown error')
   663→   663→      }
   664→   664→    }
   665→   665→
   666→   666→    // 5b: Fallback — Prisma SQLite insertion
   667→   667→    try {
   668→   668→      const db = await getPrisma()
   669→   669→
   670→   670→      const prismaMemory = await db.memory.create({
   671→   671→        data: {
   672→   672→          type: memoryType,
   673→   673→          title,
   674→   674→          content: rawContent,
   675→   675→          summary: cleanedSummary,
   676→   676→          deepInsight,
   677→   677→          sourceUrl: sourceUrl || null,
   678→   678→          tags: finalTags.join(','),
   679→   679→          imageUrl: imageUrl || null,
   680→   680→          imagePreview: imagePreviewUrl && !imageUrl ? imagePreviewUrl : null,
   681→   681→        },
   682→   682→        include: {
   683→   683→          collections: {
   684→   684→            include: {
   685→   685→              collection: {
   686→   686→                select: { id: true, name: true, color: true, icon: true },
   687→   687→              },
   688→   688→            },
   689→   689→          },
   690→   690→        },
   691→   691→      })
   692→   692→
   693→   693→      memory = buildMemoryResponse(
   694→   694→        {
   695→   695→          id: prismaMemory.id,
   696→   696→          type: prismaMemory.type,
   697→   697→          title: prismaMemory.title,
   698→   698→          content: prismaMemory.content,
   699→   699→          summary: prismaMemory.summary,
   700→   700→          deepInsight: prismaMemory.deepInsight,
   701→   701→          tags: prismaMemory.tags,
   702→   702→          sourceUrl: prismaMemory.sourceUrl,
   703→   703→          fileUrl: prismaMemory.fileUrl,
   704→   704→          imagePreview: prismaMemory.imagePreview,
   705→   705→          imageUrl: prismaMemory.imageUrl,
   706→   706→          isFavorite: prismaMemory.isFavorite,
   707→   707→          created_at: prismaMemory.createdAt.toISOString(),
   708→   708→          updated_at: prismaMemory.updatedAt.toISOString(),
   709→   709→        },
   710→   710→        prismaMemory.collections.map((mc) => ({
   711→   711→          id: mc.collection.id,
   712→   712→          name: mc.collection.name,
   713→   713→          color: mc.collection.color,
   714→   714→          icon: mc.collection.icon,
   715→   715→        }))
   716→   716→      )
   717→   717→
   718→   718→      // ── Background: embedding + link reading + auto-collections ────
   719→   719→      if (rawContent.trim()) {
   720→   720→        fetch(`${baseUrl}/api/generate-embedding`, {
   721→   721→          method: 'POST',
   722→   722→          headers: { 'Content-Type': 'application/json' },
   723→   723→          body: JSON.stringify({ memoryId: prismaMemory.id, content: rawContent.slice(0, 2000) }),
   724→   724→        }).catch(() => {})
   725→   725→      }
   726→   726→
   727→   727→      if (sourceUrl?.trim()) {
   728→   728→        fetch(`${baseUrl}/api/ai/read-link`, {
   729→   729→          method: 'POST',
   730→   730→          headers: { 'Content-Type': 'application/json' },
   731→   731→          body: JSON.stringify({ url: sourceUrl }),
   732→   732→        }).catch(() => {})
   733→   733→      }
   734→   734→
   735→   735→      // Autonomous collections (fire-and-forget)
   736→   736→      runAutonomousCollections(prismaMemory.id, finalTags, baseUrl).catch(() => {})
   737→   737→
   738→   738→      return NextResponse.json({ success: true, memory })
   739→   739→    } catch (err) {
   740→   740→      console.error('[capture] Prisma insertion failed:', err instanceof Error ? err.message : 'Unknown error')
   741→   741→      return NextResponse.json(
   742→   742→        { success: false, error: 'Failed to save memory to database' },
   743→   743→        { status: 500 }
   744→   744→      )
   745→   745→    }
   746→   746→  } catch (error) {
   747→   747→    console.error('[capture] Unhandled error:', error instanceof Error ? error.message : 'Unknown error')
   748→   748→    return NextResponse.json(
   749→   749→      { success: false, error: 'Internal server error during capture' },
   750→   750→      { status: 500 }
   751→   751→    )
   752→   752→  }
   753→   753→}
