   216→function parseCognitiveJSON(responseText: string): CognitiveResult {
   217→  // Extract JSON from response (handle markdown code blocks)
   218→  const jsonMatch = responseText.match(/\{[\s\S]*\}/)
   219→  if (jsonMatch) {
   220→    try {
   221→      const parsed = JSON.parse(jsonMatch[0])
   222→      return {
   223→        summary: typeof parsed.summary === 'string' ? parsed.summary.trim() : '',
   224→        deepInsight: typeof parsed.deep_insight === 'string' ? parsed.deep_insight.trim() : '',
   225→        tags: Array.isArray(parsed.tags)
   226→          ? parsed.tags
   227→              .filter((t: unknown) => typeof t === 'string')
   228→              .map((t: string) => t.toLowerCase().trim())
   229→              .filter(Boolean)
   230→              .slice(0, 5)
   231→          : [],
   232→      }
   233→    } catch {
   234→      // JSON parse failed — continue to regex extraction
   235→    }
   236→  }
   237→
   238→  // Fallback: try to extract individual fields via regex
   239→  const summaryMatch = responseText.match(/"summary"\s*:\s*"([^"]*)"/)
   240→  const insightMatch = responseText.match(/"deep_insight"\s*:\s*"([^"]*)"/)
   241→
   242→  return {
   243→    summary: summaryMatch?.[1]?.trim() || '',
   244→    deepInsight: insightMatch?.[1]?.trim() || '',
   245→    tags: [],
   246→  }
   247→}

   270→async function matchToExistingCollections(
   271→  memoryId: string,
   272→  memoryTags: string[],
   273→  userId: string
   274→): Promise<void> {
   275→  try {
   276→    const supabase = await getSupabaseServer()
   277→
   278→    // Fetch all user's collections
   279→    const { data: userCollections, error: collError } = await supabase
   280→      .from('collections')
   281→      .select('id, name')
   282→      .eq('user_id', userId)
   283→
   284→    if (collError || !userCollections) return
   285→
   286→    for (const collection of userCollections) {
   287→      const collNameLower = collection.name.toLowerCase()
   288→
   289→      // Check if any of the memory's tags match this collection's name
   290→      const tagMatches = memoryTags.some(
   291→        (tag) =>
   292→          collNameLower.includes(tag) ||
   293→          tag.includes(collNameLower) ||
   294→          levenshteinSimilarity(collNameLower, tag) > 0.6
   295→      )
   296→
   297→      if (tagMatches) {
   298→        // Check if this junction already exists
   299→        const { data: existingJunction } = await supabase
   300→          .from('memory_collections')
   301→          .select('id')
   302→          .eq('memory_id', memoryId)
   303→          .eq('collection_id', collection.id)
   304→          .limit(1)
   305→
   306→        if (!existingJunction || existingJunction.length === 0) {
   307→          await supabase.from('memory_collections').insert({
   308→            memory_id: memoryId,
   309→            collection_id: collection.id,
   310→          })
   311→          console.log(`[capture] Auto-assigned memory ${memoryId} to collection "${collection.name}"`)
   312→        }
   313→      }
   314→    }
   315→  } catch (err) {
   316→    console.error('[capture] Collection matching error:', err instanceof Error ? err.message : 'Unknown')
   317→  }
   318→}

   320→async function checkAutoCollectionRule(userId: string): Promise<void> {
   321→  try {
   322→    const supabase = await getSupabaseServer()
   323→
   324→    // Fetch all user's memories that have tags
   325→    const { data: allMemories, error: memError } = await supabase
   326→      .from('memories')
   327→      .select('id, tags')
   328→      .eq('user_id', userId)
   329→      .neq('tags', '')
   330→
   331→    if (memError || !allMemories || allMemories.length < 10) return
   332→
   333→    // Fetch all existing junction rows to determine which memories are collected
   334→    const { data: junctions } = await supabase
   335→      .from('memory_collections')
   336→      .select('memory_id')
   337→
   338→    const collectedMemoryIds = new Set((junctions || []).map((j) => j.memory_id))
   339→
   340→    // Filter to uncollected memories only
   341→    const uncollectedMemories = allMemories.filter((m) => !collectedMemoryIds.has(m.id))
   342→
   343→    if (uncollectedMemories.length < 10) return
   344→
   345→    // Build tag frequency map from uncollected memories
   346→    const tagFrequency: Record<string, string[]> = {}
   347→    for (const mem of uncollectedMemories) {
   348→      const memTags = (mem.tags as string)
   349→        .split(',')
   350→        .map((t) => t.trim().toLowerCase())
   351→        .filter(Boolean)
   352→      for (const tag of memTags) {
   353→        if (!tagFrequency[tag]) tagFrequency[tag] = []
   354→        tagFrequency[tag].push(mem.id)
   355→      }
   356→    }
   357→
   358→    // Fetch existing collection names to avoid duplicates
   359→    const { data: existingCollections } = await supabase
   360→      .from('collections')
   361→      .select('name')
   362→      .eq('user_id', userId)
   363→
   364→    const existingNames = new Set(
   365→      (existingCollections || []).map((c) => c.name.toLowerCase())
   366→    )
   367→
   368→    const clusterColors = [
   369→      '#6D597A', '#B56576', '#355070', '#EA526F',
   370→      '#23B5D3', '#7B2D8E', '#F18F01', '#C73E1D',
   371→      '#2D936C', '#566E3D', '#8B5CF6', '#EC4899',
   372→    ]
   373→    const { count: collectionCount } = await supabase
   374→      .from('collections')
   375→      .select('*', { count: 'exact', head: true })
   376→      .eq('user_id', userId)
   377→    let colorIdx = collectionCount || 0
   378→
   379→    // Find tag clusters that hit 10+ memories
   380→    for (const [tag, memoryIds] of Object.entries(tagFrequency)) {
   381→      if (memoryIds.length >= 10) {
   382→        const collectionName = tag.charAt(0).toUpperCase() + tag.slice(1)
   383→
   384→        // Skip if a collection with this name already exists
   385→        if (existingNames.has(collectionName.toLowerCase())) continue
   386→
   387→        const color = clusterColors[colorIdx % clusterColors.length]
   388→        colorIdx++
   389→
   390→        // Create the new collection
   391→        const { data: newCollection, error: createError } = await supabase
   392→          .from('collections')
   393→          .insert({
   394→            user_id: userId,
   395→            name: collectionName,
   396→            icon: getTagIcon(tag),
   397→            color,
   398→          })
   399→          .select('id')
   400→          .single()
   401→
   402→        if (createError || !newCollection) {
   403→          console.error('[capture] Auto-collection creation failed:', createError?.message)
   404→          continue
   405→        }
   406→
   407→        // Batch-assign all matching memories to this new collection
   408→        const junctionInserts = memoryIds.slice(0, 100).map((memId) => ({
   409→          memory_id: memId,
   410→          collection_id: newCollection.id,
   411→        }))
   412→
   413→        const { error: junctionError } = await supabase
   414→          .from('memory_collections')
   415→          .insert(junctionInserts)
   416→
   417→        if (junctionError) {
   418→          console.error('[capture] Junction insert failed:', junctionError.message)
   419→        } else {
   420→          console.log(`[capture] Auto-created collection "${collectionName}" with ${memoryIds.length} memories`)
   421→          existingNames.add(collectionName.toLowerCase())
   422→        }
   423→      }
   424→    }
   425→  } catch (err) {
   426→    console.error('[capture] Auto-collection rule error:', err instanceof Error ? err.message : 'Unknown')
   427→  }
   428→}

   451→// ─── Levenshtein similarity for fuzzy tag matching ─────────────────
   452→function levenshteinSimilarity(a: string, b: string): number {
   453→  if (a === b) return 1
   454→  if (!a.length || !b.length) return 0
   455→
   456→  const matrix: number[][] = []
   457→  for (let i = 0; i <= b.length; i++) matrix[i] = [i]
   458→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j
   459→
   460→  for (let i = 1; i <= b.length; i++) {
   461→    for (let j = 1; j <= a.length; j++) {
   462→      const cost = b[i - 1] === a[j - 1] ? 0 : 1
   463→      matrix[i][j] = Math.min(
   464→        matrix[i - 1][j] + 1,
   465→        matrix[i][j - 1] + 1,
   466→        matrix[i - 1][j - 1] + cost
   467→      )
   468→    }
   469→  }
   470→
   471→  const maxLen = Math.max(a.length, b.length)
   472→  return 1 - matrix[b.length][a.length] / maxLen
   473→}

   503→// ─── POST /api/capture — THE UNIVERSAL CAPTURE ENGINE ────────────────
   504→// ═══════════════════════════════════════════════════════════════════════
   505→export async function POST(req: NextRequest) {
   506→  try {
   507→    // ── STEP 1: MULTI-MEDIA PAYLOAD INGESTION ─────────────────────
   508→    const formData = await req.formData()
   509→
   510→    const textInput = (formData.get('text') as string) || ''
   511→    const urlInput = (formData.get('url') as string) || ''
   512→    const typeOverride = (formData.get('type') as string) || ''
   513→    const audioFile = formData.get('audio') as File | null
   514→    const imageFile = formData.get('image') as File | null
   515→
   516→    const hasText = textInput.trim().length > 0
   517→    const hasUrl = urlInput.trim().length > 0
   518→    const hasAudio = audioFile !== null && audioFile.size > 0
   519→    const hasImage = imageFile !== null && imageFile.size > 0
   520→
   521→    if (!hasText && !hasUrl && !hasAudio && !hasImage) {
   522→      return NextResponse.json(
   523→        { success: false, error: 'No content provided. Send text, audio, image, or url.' },
   524→        { status: 400 }
   525→      )
   526→    }
   527→
   528→    // ── STEP 2: AUTHENTICATE USER VIA SUPABASE ────────────────────
   529→    let supabaseUserId: string | null = null
   530→    let useSupabase = false
   531→
   532→    try {
   533→      const supabase = await getSupabaseServer()
   534→      const { data: { user: authUser }, error: authError } = await supabase.auth.getUser()
   535→      if (!authError && authUser) {
   536→        supabaseUserId = authUser.id
   537→        useSupabase = true
   538→      }
   539→    } catch {
   540→      useSupabase = false
   541→    }
   542→
   543→    // ── STEP 3: DISPATCH CHANNELS ─────────────────────────────────
   544→
   545→    // 3a: Transcribe audio if present (z-ai ASR → Groq Whisper)
   546→    let audioTranscript = ''
   547→    if (hasAudio && audioFile) {
   548→      audioTranscript = await transcribeAudio(audioFile)
   549→    }
   550→
   551→    // 3b: Upload image to Supabase Storage if present
   552→    let imageUrl: string | null = null
   553→    let imagePreviewUrl: string | null = null
   554→    if (hasImage && imageFile) {
   555→      if (useSupabase && supabaseUserId) {
   556→        imageUrl = await uploadImageToStorage(imageFile, supabaseUserId)
   557→      }
   558→      if (!imageUrl) {
   559→        // Fallback: base64 preview if storage upload fails or no Supabase
   560→        try {
   561→          const arrayBuffer = await imageFile.arrayBuffer()
   562→          const base64 = Buffer.from(arrayBuffer).toString('base64')
   563→          const mimeType = imageFile.type || 'image/png'
   564→          imagePreviewUrl = `data:${mimeType};base64,${base64}`
   565→        } catch {
   566→          imagePreviewUrl = null
   567→        }
   568→      }
   569→    }
   570→
   571→    // 3c: Determine content type
   572→    let memoryType = 'text'
   573→    if (typeOverride && ['text', 'voice', 'link', 'image'].includes(typeOverride)) {
   574→      memoryType = typeOverride
   575→    } else if (hasImage) {
   576→      memoryType = 'image'
   577→    } else if (hasAudio && !hasText && !hasUrl) {
   578→      memoryType = 'voice'
   579→    } else if (hasUrl || (hasText && /https?:\/\//.test(textInput.toLowerCase()))) {
   580→      memoryType = 'link'
   581→    }
   582→
   583→    // 3d: Compose the raw content string
   584→    let rawContent = ''
   585→    if (audioTranscript.trim()) {
   586→      rawContent = audioTranscript.trim()
   587→      if (hasText && textInput.trim()) {
   588→        rawContent = `${textInput.trim()}\n\n${audioTranscript.trim()}`
   589→      }
   590→    } else if (hasText) {
   591→      rawContent = textInput.trim()
   592→    }
   593→
   594→    const sourceUrl = hasUrl ? urlInput.trim() : memoryType === 'link' ? textInput.trim() : null
   595→
   596→    if (hasImage && !rawContent.trim()) {
   597→      rawContent = imageFile ? `Image: ${imageFile.name}` : 'Image'
   598→    }
   599→
   600→    // Compose the title
   601→    let title = rawContent.length > 80 ? rawContent.slice(0, 80) + '...' : rawContent
   602→    if (hasImage && !textInput.trim()) {
   603→      title = imageFile ? `Image: ${imageFile.name}` : 'Image'
   604→    }
   605→
   606→    // ── STEP 4: HIGH-END COGNITIVE SYNTHESIS (3-TIER AI PIPELINE) ─
   607→    const textForSynthesis = rawContent || (hasUrl ? urlInput : '')
   608→    let cleanedSummary: string | null = null
   609→    let deepInsight: string | null = null
   610→    let aiTags: string[] = []
   611→
   612→    // Always run keyword tags first (instant, free, reliable)
   613→    const keywordTags = autoGenerateTags(rawContent, title)
   614→
   615→    // Run AI synthesis if we have content (with 10-second timeout)
   616→    if (textForSynthesis.trim()) {
   617→      try {
   618→        const synthesisPromise = generateCognitiveSynthesis(textForSynthesis)
   619→        const timeoutPromise = new Promise<CognitiveResult>((resolve) =>
   620→          setTimeout(() => resolve({ summary: '', deepInsight: '', tags: [] }), 10000)
   621→        )
   622→
   623→        const cognitive = await Promise.race([synthesisPromise, timeoutPromise])
   624→        if (cognitive.summary) cleanedSummary = cognitive.summary
   625→        if (cognitive.deepInsight) deepInsight = cognitive.deepInsight
   626→        if (cognitive.tags.length > 0) aiTags = cognitive.tags
   627→      } catch (err) {
   628→        console.error('[capture] Synthesis error:', err instanceof Error ? err.message : 'Unknown')
   629→      }
   630→    }
   631→
   632→    // Merge: AI tags take priority, keyword tags fill gaps
   633→    const finalTags = aiTags.length > 0
   634→      ? [...new Set([...aiTags, ...keywordTags])].slice(0, 5)
   635→      : keywordTags
   636→
   637→    // ── STEP 5: DATABASE INGESTION & AUTO-COLLECTIONS ─────────────
   638→    let memory: Record<string, unknown>
   639→
   640→    // 5a: Supabase-first insertion (authenticated users with RLS)
   641→    if (useSupabase && supabaseUserId) {
   642→      try {
   643→        const supabase = await getSupabaseServer()
   644→        const insertData: Record<string, unknown> = {
   645→          user_id: supabaseUserId,
   646→          type: memoryType,
   647→          title,
   648→          content: rawContent,
   649→          source_url: sourceUrl || null,
   650→          tags: finalTags.join(','),
   651→        }
   652→
   653→        if (cleanedSummary) insertData.summary = cleanedSummary
   654→        if (deepInsight) insertData.deep_insight = deepInsight
   655→        if (imageUrl) insertData.image_url = imageUrl
   656→        if (imagePreviewUrl && !imageUrl) insertData.image_preview = imagePreviewUrl
   657→
   658→        const { data: memoryRow, error: insertError } = await supabase
   659→          .from('memories')
   660→          .insert(insertData)
   661→          .select('*, memory_collections(collection_id, collections(id, name, color, icon))')
   662→          .single()
   663→
   664→        if (insertError) throw insertError
   665→
   666→        const row = memoryRow as Record<string, unknown>
   667→        const collectionsFromRow = (
   668→          (row.memory_collections as Record<string, Record<string, unknown>>[]) || []
   669→        ).map((mc) => ({
   670→          id: (mc.collections as Record<string, unknown>)?.id as string,
   671→          name: (mc.collections as Record<string, unknown>)?.name as string,
   672→          color: (mc.collections as Record<string, unknown>)?.color as string,
   673→          icon: (mc.collections as Record<string, unknown>)?.icon as string,
   674→        }))
   675→
   676→        memory = buildMemoryResponse(row, collectionsFromRow)
   677→
   678→        // Background: autonomous collections engine
   679→        runAutonomousCollections(memory.id as string, finalTags, supabaseUserId).catch(() => {})
   680→
   681→        return NextResponse.json({ success: true, memory })
   682→      } catch (err) {
   683→        console.error('[capture] Supabase insertion failed, falling back to Prisma:', err instanceof Error ? err.message : 'Unknown')
   684→      }
   685→    }
   686→
   687→    // 5b: Fallback — Prisma SQLite insertion (local / no Supabase)
   688→    try {
   689→      const { db } = await import('@/lib/db')
   690→
   691→      const prismaMemory = await db.memory.create({
   692→        data: {
   693→          type: memoryType,
   694→          title,
   695→          content: rawContent,
   696→          summary: cleanedSummary,
   697→          deepInsight,
   698→          sourceUrl: sourceUrl || null,
   699→          tags: finalTags.join(','),
   700→          imageUrl: imageUrl || null,
   701→          imagePreview: imagePreviewUrl && !imageUrl ? imagePreviewUrl : null,
   702→        },
   703→        include: {
   704→          collections: {
   705→            include: {
   706→              collection: {
   707→                select: { id: true, name: true, color: true, icon: true },
   708→              },
   709→            },
   710→          },
   711→        },
   712→      })
   713→
   714→      memory = buildMemoryResponse(
   715→        {
   716→          id: prismaMemory.id,
   717→          type: prismaMemory.type,
   718→          title: prismaMemory.title,
   719→          content: prismaMemory.content,
   720→          summary: prismaMemory.summary,
   721→          deepInsight: prismaMemory.deepInsight,
   722→          tags: prismaMemory.tags,
   723→          sourceUrl: prismaMemory.sourceUrl,
   724→          fileUrl: prismaMemory.fileUrl,
   725→          imagePreview: prismaMemory.imagePreview,
   726→          imageUrl: prismaMemory.imageUrl,
   727→          isFavorite: prismaMemory.isFavorite,
   728→          created_at: prismaMemory.createdAt.toISOString(),
   729→          updated_at: prismaMemory.updatedAt.toISOString(),
   730→        },
   731→        prismaMemory.collections.map((mc) => ({
   732→          id: mc.collection.id,
   733→          name: mc.collection.name,
   734→          color: mc.collection.color,
   735→          icon: mc.collection.icon,
   736→        }))
   737→      )
   738→
   739→      return NextResponse.json({ success: true, memory })
   740→    } catch (err) {
   741→      console.error('[capture] Prisma insertion failed:', err instanceof Error ? err.message : 'Unknown')
   742→      return NextResponse.json(
   743→        { success: false, error: 'Failed to save memory to database' },
   744→        { status: 500 }
   745→      )
   746→    }
   747→
   748→  } catch (error) {
   749→    console.error('[capture] Unhandled error:', error instanceof Error ? error.message : 'Unknown')
   750→    return NextResponse.json(
   751→      { success: false, error: 'Internal server error during capture' },
   752→      { status: 500 }
   753→    )
   754→  }
   755→}

   757→// ═══════════════════════════════════════════════════════════════════════
   758→// ─── DELETE /api/capture — PURGE MEMORY FROM SUPABASE ────────────────
   759→// ═══════════════════════════════════════════════════════════════════════
   760→export async function DELETE(req: NextRequest) {
   761→  try {
   762→    const { searchParams } = new URL(req.url)
   763→    const memoryId = searchParams.get('id')
   764→
   765→    if (!memoryId) {
   766→      return NextResponse.json(
   767→        { success: false, error: 'Memory ID is required' },
   768→        { status: 400 }
   769→      )
   770→    }
   771→
   772→    // Try Supabase deletion first
   773→    let supabaseDeleted = false
   774→    try {
   775→      const supabase = await getSupabaseServer()
   776→      const { data: { user: authUser }, error: authError } = await supabase.auth.getUser()
   777→
   778→      if (!authError && authUser) {
   779→        // Delete junction rows first (foreign key constraint)
   780→        const { error: junctionError } = await supabase
   781→          .from('memory_collections')
   782→          .delete()
   783→          .eq('memory_id', memoryId)
   784→
   785→        if (junctionError) {
   786→          console.error('[capture] Junction delete error:', junctionError.message)
   787→        }
   788→
   789→        // Delete the memory itself
   790→        const { error: deleteError } = await supabase
   791→          .from('memories')
   792→          .delete()
   793→          .eq('id', memoryId)
   794→          .eq('user_id', authUser.id)
   795→
   796→        if (!deleteError) {
   797→          supabaseDeleted = true
   798→        } else {
   799→          console.error('[capture] Supabase delete failed:', deleteError.message)
   800→        }
   801→      }
   802→    } catch {
   803→      // Supabase not available — fall through to Prisma
   804→    }
   805→
   806→    // Fallback: Prisma deletion (always attempt if Supabase didn't succeed)
   807→    if (!supabaseDeleted) {
   808→      try {
   809→        const { db } = await import('@/lib/db')
   810→        await db.memory.delete({ where: { id: memoryId } })
   811→      } catch (prismaErr) {
   812→        console.error('[capture] Prisma delete failed:', prismaErr instanceof Error ? prismaErr.message : 'Unknown')
   813→        if (!supabaseDeleted) {
   814→          return NextResponse.json(
   815→            { success: false, error: 'Failed to delete memory' },
   816→            { status: 500 }
   817→          )
   818→        }
   819→      }
   820→    }
   821→
   822→    return NextResponse.json({ success: true })
   823→
   824→  } catch (error) {
   825→    console.error('[capture] Delete error:', error instanceof Error ? error.message : 'Unknown')
   826→    return NextResponse.json(
   827→      { success: false, error: 'Internal server error during deletion' },
   828→      { status: 500 }
   829→    )
   830→  }
   831→}
