    50→async function transcribeAudio(audioFile: File): Promise<string> {
    51→  // TIER 1: z-ai-web-dev-sdk ASR
    52→  try {
    53→    const { createTranscription } = await import('@/lib/aether-asr')

   209→function parseCognitiveJSON(responseText: string): CognitiveResult {
   210→  // Extract JSON from response (handle markdown code blocks)
   211→  const jsonMatch = responseText.match(/\{[\s\S]*\}/)
   212→  if (jsonMatch) {
   213→    try {
   214→      const parsed = JSON.parse(jsonMatch[0])
   215→      return {
   216→        summary: typeof parsed.summary === 'string' ? parsed.summary.trim() : '',
   217→        deepInsight: typeof parsed.deep_insight === 'string' ? parsed.deep_insight.trim() : '',
   218→        tags: Array.isArray(parsed.tags)
   219→          ? parsed.tags.filter((t: unknown) => typeof t === 'string').map((t: string) => t.toLowerCase().trim()).filter(Boolean).slice(0, 5)
   220→          : [],
   221→      }
   222→    } catch {
   223→      // JSON parse failed
   224→    }
   225→  }
   226→
   227→  // Fallback: try to extract individual fields
   228→  const summaryMatch = responseText.match(/"summary"\s*:\s*"([^"]*)"/)
   229→  const insightMatch = responseText.match(/"deep_insight"\s*:\s*"([^"]*)"/)
   230→
   231→  return {
   232→    summary: summaryMatch?.[1]?.trim() || '',
   233→    deepInsight: insightMatch?.[1]?.trim() || '',
   234→    tags: [],
   235→  }
   236→}

   257→async function matchToExistingCollections(memoryId: string, memoryTags: string[]): Promise<void> {
   258→  try {
   259→    const db = await getPrisma()
   260→    const collections = await db.collection.findMany()
   261→
   262→    for (const collection of collections) {
   263→      const collNameLower = collection.name.toLowerCase()
   264→      const tagMatches = memoryTags.some(
   265→        (tag) =>
   266→          collNameLower.includes(tag) ||
   267→          tag.includes(collNameLower) ||
   268→          levenshteinSimilarity(collNameLower, tag) > 0.6
   269→      )
   270→
   271→      if (tagMatches) {
   272→        const existing = await db.memoryCollection.findFirst({
   273→          where: { memoryId, collectionId: collection.id },
   274→        })
   275→
   276→        if (!existing) {
   277→          await db.memoryCollection.create({
   278→            data: { memoryId, collectionId: collection.id },
   279→          })
   280→        }
   281→      }
   282→    }
   283→
   284→    // Also try Supabase path
   285→    try {
   286→      const supabase = await getSupabaseServer()
   287→      const { data: { user } } = await supabase.auth.getUser()
   288→      if (user) {
   289→        const { data: supabaseCollections } = await supabase
   290→          .from('collections')
   291→          .select('id, name')
   292→          .eq('user_id', user.id)
   293→
   294→        if (supabaseCollections) {
   295→          for (const col of supabaseCollections) {
   296→            const colNameLower = col.name.toLowerCase()
   297→            const tagMatches = memoryTags.some(
   298→              (tag) =>
   299→                colNameLower.includes(tag) ||
   300→                tag.includes(colNameLower) ||
   301→                levenshteinSimilarity(colNameLower, tag) > 0.6
   302→            )
   303→            if (tagMatches) {
   304→              await supabase.from('memory_collections').upsert({
   305→                memory_id: memoryId,
   306→                collection_id: col.id,
   307→              }, { onConflict: 'memory_id,collection_id' })
   308→            }
   309→          }
   310→        }
   311→      }
   312→    } catch {
   313→      // Supabase path optional
   314→    }
   315→  } catch (err) {
   316→    console.error('[capture] Collection matching error:', err instanceof Error ? err.message : 'Unknown')
   317→  }
   318→}

   323→
   324→    const uncollectedMemories = await db.memory.findMany({

   326→        collections: { none: {} },
   327→        tags: { not: '' },
   328→      },
   329→      select: { id: true, tags: true },
   330→    })
   331→
   332→    if (uncollectedMemories.length < 10) return
   333→
   334→    // Build tag frequency map
   335→    const tagFrequency: Record<string, string[]> = {}
   336→    for (const mem of uncollectedMemories) {
   337→      const memTags = mem.tags.split(',').map((t) => t.trim().toLowerCase()).filter(Boolean)
   338→      for (const tag of memTags) {
   339→        if (!tagFrequency[tag]) tagFrequency[tag] = []
   340→        tagFrequency[tag].push(mem.id)
   341→      }
   342→    }
   343→
   344→    // Find tag clusters that hit 10+ memories
   345→    const clusterColors = ['#6D597A', '#B56576', '#355070', '#EA526F', '#23B5D3', '#7B2D8E', '#F18F01', '#C73E1D']
   346→    let colorIdx = await db.collection.count()
   347→
   348→    for (const [tag, memoryIds] of Object.entries(tagFrequency)) {
   349→      if (memoryIds.length >= 10) {
   350→        const existingCollection = await db.collection.findFirst({
   351→          where: { name: { equals: tag.charAt(0).toUpperCase() + tag.slice(1), mode: 'insensitive' } },

   353→
   354→        if (!existingCollection) {
   355→          const collectionName = tag.charAt(0).toUpperCase() + tag.slice(1)

   361→              name: collectionName,

   373→                memoryId: memId,

   375→              },

   388→                .from('collections')
   389→                .insert({
   390→                  user_id: user.id,
   391→                  name: collectionName,
   392→                  icon: getTagIcon(tag),
   393→                  color,
   394→                })
   395→                .select('id')
   396→                .single()
   397→
   398→              if (supaCol) {
   399→                for (const memId of memoryIds.slice(0, 50)) {
   400→                  await supabase.from('memory_collections').upsert({
   401→                    memory_id: memId,

   410→        }

   412→    }

   429→// ─── Levenshtein similarity for fuzzy tag matching ─────────────────
   430→function levenshteinSimilarity(a: string, b: string): number {
   431→  if (a === b) return 1
   432→  if (!a.length || !b.length) return 0
   433→
   434→  const matrix: number[][] = []
   435→  for (let i = 0; i <= b.length; i++) matrix[i] = [i]
   436→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j
   437→
   438→  for (let i = 1; i <= b.length; i++) {
   439→    for (let j = 1; j <= a.length; j++) {
   440→      const cost = b[i - 1] === a[j - 1] ? 0 : 1
   441→      matrix[i][j] = Math.min(
   442→        matrix[i - 1][j] + 1,
   443→        matrix[i][j - 1] + 1,
   444→        matrix[i - 1][j - 1] + cost
   445→      )
   446→    }
   447→  }
   448→
   449→  const maxLen = Math.max(a.length, b.length)
   450→  return 1 - matrix[b.length][a.length] / maxLen
   451→}

   504→    }
   505→
   506→    // ── STEP 2: Determine user (Supabase session or local) ─────────
   507→    let supabaseSession: { user: { id: string } } | null = null
   508→    let useSupabase = false
   509→
   510→    try {
   511→      const supabase = await getSupabaseServer()
   512→      const { data: { user }, error: authError } = await supabase.auth.getUser()
   513→      if (!authError && user) {
   514→        supabaseSession = { user }
   515→        const { error: tableCheck } = await supabase
   516→          .from('memories')
   517→          .select('id')
   518→          .eq('user_id', user.id)
   519→          .limit(1)
   520→        useSupabase = !tableCheck
   521→      }
   522→    } catch {
   523→      // No Supabase session — fall back to Prisma
   524→    }
   525→
   526→    // ── STEP 3: DYNAMIC INPUT DISPATCHING ────────────────────────────
   527→
   528→    // 3a: Transcribe audio if present
   529→    let audioTranscript = ''
   530→    if (hasAudio && audioFile) {
   531→      audioTranscript = await transcribeAudio(audioFile)
   532→    }
   533→
   534→    // 3b: Upload image if present
   535→    let imageUrl: string | null = null
   536→    let imagePreviewUrl: string | null = null
   537→    if (hasImage && imageFile) {
   538→      if (useSupabase && supabaseSession) {
   539→        imageUrl = await uploadImageToStorage(imageFile, supabaseSession.user.id)
   540→      }
   541→      if (!imageUrl) {
   542→        try {
   543→          const arrayBuffer = await imageFile.arrayBuffer()
   544→          const base64 = Buffer.from(arrayBuffer).toString('base64')
   545→          const mimeType = imageFile.type || 'image/png'
   546→          imagePreviewUrl = `data:${mimeType};base64,${base64}`
   547→        } catch {
   548→          imagePreviewUrl = null
   549→        }
   550→      }
   551→    }
   552→
   553→    // 3c: Determine content type
   554→    let memoryType = 'text'
   555→    if (typeOverride && ['text', 'voice', 'link', 'image'].includes(typeOverride)) {
   556→      memoryType = typeOverride
   557→    } else if (hasImage) {
   558→      memoryType = 'image'
   559→    } else if (hasAudio && !hasText && !hasUrl) {
   560→      memoryType = 'voice'
   561→    } else if (hasUrl || (hasText && /https?:\/\//.test(textInput.toLowerCase()))) {
   562→      memoryType = 'link'
   563→    }
   564→
   565→    // 3d: Compose the raw content string
   566→    let rawContent = ''
   567→    if (audioTranscript.trim()) {
   568→      rawContent = audioTranscript.trim()
   569→      if (hasText && textInput.trim()) {
   570→        rawContent = `${textInput.trim()}\n\n${audioTranscript.trim()}`
   571→      }
   572→    } else if (hasText) {
   573→      rawContent = textInput.trim()
   574→    }
   575→
   576→    const sourceUrl = hasUrl ? urlInput.trim() : memoryType === 'link' ? textInput.trim() : null
   577→
   578→    if (hasImage && !rawContent.trim()) {
   579→      rawContent = imageFile ? `Image: ${imageFile.name}` : 'Image'
   580→    }
   581→
   582→    // Compose the title
   583→    let title = rawContent.length > 80 ? rawContent.slice(0, 80) + '...' : rawContent
   584→    if (hasImage && !textInput.trim()) {
   585→      title = imageFile ? `Image: ${imageFile.name}` : 'Image'
   586→    }
   587→
   588→    // ── STEP 4: COGNITIVE SYNTHESIS — 3-TIER AI PIPELINE ───────────
   589→    const textForSynthesis = rawContent || (hasUrl ? urlInput : '')
   590→    let cleanedSummary: string | null = null
   591→    let deepInsight: string | null = null
   592→    let aiTags: string[] = []
   593→
   594→    // Always run keyword tags first (instant, free, reliable)
   595→    const keywordTags = autoGenerateTags(rawContent, title)
   596→
   597→    // Run AI synthesis if we have content (with 8-second timeout)
   598→    if (textForSynthesis.trim()) {
   599→      try {
   600→        const synthesisPromise = generateCognitiveSynthesis(textForSynthesis)
   601→        const timeoutPromise = new Promise<CognitiveResult>((resolve) =>
   602→          setTimeout(() => resolve({ summary: '', deepInsight: '', tags: [] }), 8000)
   603→        )
   604→
   605→        const cognitive = await Promise.race([synthesisPromise, timeoutPromise])
   606→        if (cognitive.summary) cleanedSummary = cognitive.summary
   607→        if (cognitive.deepInsight) deepInsight = cognitive.deepInsight
   608→        if (cognitive.tags.length > 0) aiTags = cognitive.tags
   609→      } catch (err) {
   610→        console.error('[capture] Synthesis error:', err instanceof Error ? err.message : 'Unknown')
   611→      }
   612→    }
   613→
   614→    // Merge: AI tags take priority, keyword tags fill gaps
   615→    const finalTags = aiTags.length > 0
   616→      ? [...new Set([...aiTags, ...keywordTags])].slice(0, 5)
   617→      : keywordTags
   618→
   619→    // ── STEP 5: DATABASE INGESTION ───────────────────────────────────
   620→    let memory: Record<string, unknown>
   621→
   622→    // 5a: Try Supabase insertion
   623→    if (useSupabase && supabaseSession) {
   624→      try {
   625→        const supabase = await getSupabaseServer()
   626→        const insertData: Record<string, unknown> = {
   627→          user_id: supabaseSession.user.id,
   628→          type: memoryType,
   629→          title,
   630→          content: rawContent,
   631→          source_url: sourceUrl || null,
   632→          tags: finalTags.join(','),
   633→        }
   634→
   635→        if (cleanedSummary) insertData.summary = cleanedSummary
   636→        if (deepInsight) insertData.deep_insight = deepInsight
   637→        if (imageUrl) insertData.image_url = imageUrl
   638→        if (imagePreviewUrl && !imageUrl) insertData.image_preview = imagePreviewUrl
   639→
   640→        const { data: memoryRow, error: insertError } = await supabase
   641→          .from('memories')
   642→          .insert(insertData)
   643→          .select('*, memory_collections(collection_id, collections(id, name, color, icon))')
   644→          .single()
   645→
   646→        if (insertError) throw insertError
   647→
   648→        const row = memoryRow as Record<string, unknown>
   649→        const collectionsFromRow = (
   650→          (row.memory_collections as Record<string, Record<string, unknown>>[]) || []
   651→        ).map((mc) => ({
   652→          id: (mc.collections as Record<string, unknown>)?.id as string,
   653→          name: (mc.collections as Record<string, unknown>)?.name as string,
   654→          color: (mc.collections as Record<string, unknown>)?.color as string,
   655→          icon: (mc.collections as Record<string, unknown>)?.icon as string,
   656→        }))
   657→
   658→        memory = buildMemoryResponse(row, collectionsFromRow)
   659→
   660→        // Background: embedding + link reading + auto-collections
   661→        if (rawContent.trim()) {
   662→          fetch(`${baseUrl}/api/generate-embedding`, {
   663→            method: 'POST',
   664→            headers: { 'Content-Type': 'application/json' },
   665→            body: JSON.stringify({ memoryId: memory.id, content: rawContent.slice(0, 2000) }),
   666→          }).catch(() => {})
   667→        }
   668→
   669→        if (sourceUrl?.trim()) {
   670→          fetch(`${baseUrl}/api/ai/read-link`, {
   671→            method: 'POST',
   672→            headers: { 'Content-Type': 'application/json' },
   673→            body: JSON.stringify({ url: sourceUrl }),
   674→          }).catch(() => {})
   675→        }
   676→
   677→        runAutonomousCollections(memory.id as string, finalTags, baseUrl).catch(() => {})
   678→
   679→        return NextResponse.json({ success: true, memory })
   680→      } catch (err) {
   681→        console.error('[capture] Supabase insertion failed, falling back to Prisma:', err instanceof Error ? err.message : 'Unknown')
   682→      }
   683→    }
   684→
   685→    // 5b: Fallback — Prisma SQLite insertion (always works)
   686→    try {
   687→      const db = await getPrisma()
   688→
   689→      const prismaMemory = await db.memory.create({
   690→        data: {
   691→          type: memoryType,
   692→          title,
   693→          content: rawContent,
   694→          summary: cleanedSummary,
   695→          deepInsight,
   696→          sourceUrl: sourceUrl || null,
   697→          tags: finalTags.join(','),
   698→          imageUrl: imageUrl || null,
   699→          imagePreview: imagePreviewUrl && !imageUrl ? imagePreviewUrl : null,
   700→        },
   701→        include: {
   702→          collections: {
   703→            include: {
   704→              collection: {
   705→                select: { id: true, name: true, color: true, icon: true },
   706→              },
   707→            },
   708→          },
   709→        },
   710→      })
   711→
   712→      memory = buildMemoryResponse(
   713→        {
   714→          id: prismaMemory.id,
   715→          type: prismaMemory.type,
   716→          title: prismaMemory.title,
   717→          content: prismaMemory.content,
   718→          summary: prismaMemory.summary,
   719→          deepInsight: prismaMemory.deepInsight,
   720→          tags: prismaMemory.tags,
   721→          sourceUrl: prismaMemory.sourceUrl,
   722→          fileUrl: prismaMemory.fileUrl,
   723→          imagePreview: prismaMemory.imagePreview,
   724→          imageUrl: prismaMemory.imageUrl,
   725→          isFavorite: prismaMemory.isFavorite,
   726→          created_at: prismaMemory.createdAt.toISOString(),
   727→          updated_at: prismaMemory.updatedAt.toISOString(),
   728→        },
   729→        prismaMemory.collections.map((mc) => ({
   730→          id: mc.collection.id,
   731→          name: mc.collection.name,
   732→          color: mc.collection.color,
   733→          icon: mc.collection.icon,
   734→        }))
   735→      )
   736→
   737→      // Background: embedding + link reading + auto-collections
   738→      if (rawContent.trim()) {
   739→        fetch(`${baseUrl}/api/generate-embedding`, {
   740→          method: 'POST',
   741→          headers: { 'Content-Type': 'application/json' },
   742→          body: JSON.stringify({ memoryId: prismaMemory.id, content: rawContent.slice(0, 2000) }),
   743→        }).catch(() => {})
   744→      }
   745→
   746→      if (sourceUrl?.trim()) {
   747→        fetch(`${baseUrl}/api/ai/read-link`, {
   748→          method: 'POST',
   749→          headers: { 'Content-Type': 'application/json' },
   750→          body: JSON.stringify({ url: sourceUrl }),
   751→        }).catch(() => {})
   752→      }
   753→
   754→      runAutonomousCollections(prismaMemory.id, finalTags, baseUrl).catch(() => {})
   755→
   756→      return NextResponse.json({ success: true, memory })
   757→    } catch (err) {
   758→      console.error('[capture] Prisma insertion failed:', err instanceof Error ? err.message : 'Unknown')
   759→      return NextResponse.json(
   760→        { success: false, error: 'Failed to save memory to database' },
   761→        { status: 500 }
   762→      )
   763→    }
   764→  } catch (error) {
   765→    console.error('[capture] Unhandled error:', error instanceof Error ? error.message : 'Unknown')
   766→    return NextResponse.json(
   767→      { success: false, error: 'Internal server error during capture' },
   768→      { status: 500 }
   769→    )
   770→  }
   771→}
