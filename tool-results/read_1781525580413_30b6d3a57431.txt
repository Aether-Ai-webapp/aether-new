   430вҶ’   430вҶ’function levenshteinSimilarity(a: string, b: string): number {
   431вҶ’   431вҶ’  if (a === b) return 1
   432вҶ’   432вҶ’  if (!a.length || !b.length) return 0
   433вҶ’   433вҶ’
   434вҶ’   434вҶ’  const matrix: number[][] = []
   435вҶ’   435вҶ’  for (let i = 0; i <= b.length; i++) matrix[i] = [i]
   436вҶ’   436вҶ’  for (let j = 0; j <= a.length; j++) matrix[0][j] = j
   437вҶ’   437вҶ’
   438вҶ’   438вҶ’  for (let i = 1; i <= b.length; i++) {
   439вҶ’   439вҶ’    for (let j = 1; j <= a.length; j++) {
   440вҶ’   440вҶ’      const cost = b[i - 1] === a[j - 1] ? 0 : 1
   441вҶ’   441вҶ’      matrix[i][j] = Math.min(
   442вҶ’   442вҶ’        matrix[i - 1][j] + 1,
   443вҶ’   443вҶ’        matrix[i][j - 1] + 1,
   444вҶ’   444вҶ’        matrix[i - 1][j - 1] + cost
   445вҶ’   445вҶ’      )
   446вҶ’   446вҶ’    }
   447вҶ’   447вҶ’  }
   448вҶ’   448вҶ’
   449вҶ’   449вҶ’  const maxLen = Math.max(a.length, b.length)
   450вҶ’   450вҶ’  return 1 - matrix[b.length][a.length] / maxLen
   451вҶ’   451вҶ’}
