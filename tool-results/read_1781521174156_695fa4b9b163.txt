   117→   117→// ─── Smart Token Saver: Local, free, instant tagging ──────────────────
   118→   118→function getLocalTags(text: string): string[] {
   119→   119→  const lower = text.toLowerCase()
   120→   120→  const tags: string[] = []
   121→   121→
   122→   122→  // URL detection → 'link'
   123→   123→  if (/https?:\/\//.test(lower)) {
   124→   124→    tags.push('link')
   125→   125→  }
   126→   126→
   127→   127→  // Task words → 'task'
   128→   128→  if (/\b(todo|remind|need to|must|buy)\b/.test(lower)) {
   129→   129→    tags.push('task')
   130→   130→  }
   131→   131→
   132→   132→  // Person/personal words → 'personal'
   133→   133→  if (/\b(i |me |with |call|email)\b/.test(lower)) {
   134→   134→    tags.push('personal')
   135→   135→  }
   136→   136→
   137→   137→  // Question/idea words → 'idea'
   138→   138→  if (/\b(what|how|why|who)\b/.test(lower)) {
   139→   139→    tags.push('idea')
   140→   140→  }
   141→   141→
   142→   142→  return tags.slice(0, 5)
   143→   143→}

   145→   145→// ─── Extended keyword-based auto-tagging (richer, for AI fallback) ───
   146→   146→function autoGenerateTags(content: string, title: string): string[] {
   147→   147→  const text = `${title} ${content}`.toLowerCase()
   148→   148→  const tagMap: Record<string, string[]> = {
   149→   149→    'work': ['meeting', 'project', 'deadline', 'client', 'office', 'team', 'q1', 'q2', 'q3', 'q4', 'quarterly', 'strategy'],
   150→   150→    'personal': ['routine', 'morning', 'exercise', 'meditation', 'journal', 'habit'],
   151→   151→    'travel': ['trip', 'itinerary', 'flight', 'hotel', 'visit', 'tokyo', 'paris', 'destination'],
   152→   152→    'learning': ['learn', 'study', 'course', 'tutorial', 'book', 'read', 'article'],
   153→   153→    'code': ['code', 'programming', 'react', 'javascript', 'typescript', 'api', 'bug', 'feature', 'css', 'html', 'framework'],
   154→   154→    'design': ['design', 'ui', 'ux', 'layout', 'color', 'font', 'figma', 'wireframe'],
   155→   155→    'ai': ['ai', 'machine learning', 'neural', 'model', 'gpt', 'gemini', 'llm', 'chatbot'],
   156→   156→    'recipe': ['recipe', 'cook', 'bake', 'ingredient', 'food', 'meal', 'breakfast', 'dinner'],
   157→   157→    'idea': ['idea', 'concept', 'brainstorm', 'innovative', 'startup', 'prototype'],
   158→   158→    'finance': ['budget', 'expense', 'invest', 'savings', 'money', 'cost'],
   159→   159→  }
   160→   160→
   161→   161→  const tags: string[] = []
   162→   162→  for (const [tag, keywords] of Object.entries(tagMap)) {
   163→   163→    if (keywords.some(kw => text.includes(kw))) {
   164→   164→      tags.push(tag)
   165→   165→    }
   166→   166→  }
   167→   167→  return tags.slice(0, 5)
   168→   168→}

   256→   256→export const useAetherStore = create<AetherState>((set, get) => ({

   268→   268→  setShowAuthModal: (showAuthModal) => set({ showAuthModal }),
   269→   269→
   270→   270→  // The gate: if user is signed in, run action immediately. Otherwise, queue it and show modal.
   271→   271→  requireAuth: (action) => {
   272→   272→    const state = get()
   273→   273→    if (state.isAuthenticated && state.user && state.user.id !== 'local') {
   274→   274→      action()

   276→   276→      set({ showAuthModal: true, pendingAction: action })
   277→   277→    }

   280→   280→  login: async (email, password) => {
   281→   281→    try {
   282→   282→      const res = await fetch('/api/auth/login', {
   283→   283→        method: 'POST',
   284→   284→        headers: { 'Content-Type': 'application/json' },
   285→   285→        body: JSON.stringify({ email, password }),
   286→   286→      })
   287→   287→      if (!res.ok) return false
   288→   288→      const data = await res.json()
   289→   289→      if (data.user) {
   290→   290→        set({
   291→   291→          user: {
   292→   292→            id: data.user.id,
   293→   293→            email: data.user.email || '',
   294→   294→            name: data.user.user_metadata?.name || '',
   295→   295→            avatarUrl: data.user.user_metadata?.avatar_url || null,
   296→   296→          },
   297→   297→          isAuthenticated: true,
   298→   298→          showAuthModal: false,
   299→   299→        })
   300→   300→        // Check if Supabase tables exist
   301→   301→        get().checkSupabaseTables()
   302→   302→        // Execute any pending action that was queued
   303→   303→        const pending = get().pendingAction
   304→   304→        if (pending) {
   305→   305→          pending()
   306→   306→          set({ pendingAction: null })
   307→   307→        }
   308→   308→        // Re-fetch data from Supabase
   309→   309→        get().fetchMemories()
   310→   310→        get().fetchCollections()
   311→   311→        return true
   312→   312→      }
   313→   313→      return false
   314→   314→    } catch {
   315→   315→      return false
   316→   316→    }
   317→   317→  },
   318→   318→
