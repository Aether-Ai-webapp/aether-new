   108→   108→// ─── Smart Token Saver: Local, free, instant tagging ──────────────────
   109→   109→function getLocalTags(text: string): string[] {
   110→   110→  const lower = text.toLowerCase()
   111→   111→  const tags: string[] = []
   112→   112→
   113→   113→  // URL detection → 'link'
   114→   114→  if (/https?:\/\//.test(lower)) {
   115→   115→    tags.push('link')
   116→   116→  }
   117→   117→
   118→   118→  // Task words → 'task'
   119→   119→  if (/\b(todo|remind|need to|must|buy)\b/.test(lower)) {
   120→   120→    tags.push('task')
   121→   121→  }
   122→   122→
   123→   123→  // Person/personal words → 'personal'
   124→   124→  if (/\b(i |me |with |call|email)\b/.test(lower)) {
   125→   125→    tags.push('personal')
   126→   126→  }
   127→   127→
   128→   128→  // Question/idea words → 'idea'
   129→   129→  if (/\b(what|how|why|who)\b/.test(lower)) {
   130→   130→    tags.push('idea')
   131→   131→  }
   132→   132→
   133→   133→  return tags.slice(0, 5)
   134→   134→}

   136→   136→// ─── Extended keyword-based auto-tagging (richer, for AI fallback) ───
   137→   137→function autoGenerateTags(content: string, title: string): string[] {
   138→   138→  const text = `${title} ${content}`.toLowerCase()
   139→   139→  const tagMap: Record<string, string[]> = {
   140→   140→    'work': ['meeting', 'project', 'deadline', 'client', 'office', 'team', 'q1', 'q2', 'q3', 'q4', 'quarterly', 'strategy'],
   141→   141→    'personal': ['routine', 'morning', 'exercise', 'meditation', 'journal', 'habit'],
   142→   142→    'travel': ['trip', 'itinerary', 'flight', 'hotel', 'visit', 'tokyo', 'paris', 'destination'],
   143→   143→    'learning': ['learn', 'study', 'course', 'tutorial', 'book', 'read', 'article'],
   144→   144→    'code': ['code', 'programming', 'react', 'javascript', 'typescript', 'api', 'bug', 'feature', 'css', 'html', 'framework'],
   145→   145→    'design': ['design', 'ui', 'ux', 'layout', 'color', 'font', 'figma', 'wireframe'],
   146→   146→    'ai': ['ai', 'machine learning', 'neural', 'model', 'gpt', 'gemini', 'llm', 'chatbot'],
   147→   147→    'recipe': ['recipe', 'cook', 'bake', 'ingredient', 'food', 'meal', 'breakfast', 'dinner'],
   148→   148→    'idea': ['idea', 'concept', 'brainstorm', 'innovative', 'startup', 'prototype'],
   149→   149→    'finance': ['budget', 'expense', 'invest', 'savings', 'money', 'cost'],
   150→   150→  }
   151→   151→
   152→   152→  const tags: string[] = []
   153→   153→  for (const [tag, keywords] of Object.entries(tagMap)) {
   154→   154→    if (keywords.some(kw => text.includes(kw))) {
   155→   155→      tags.push(tag)
   156→   156→    }
   157→   157→  }
   158→   158→  return tags.slice(0, 5)
   159→   159→}
