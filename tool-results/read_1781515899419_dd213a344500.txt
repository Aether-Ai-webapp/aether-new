    78→function mapSupabaseMemory(row: SupabaseMemoryRow): Memory {
    79→  return {
    80→    id: row.id,
    81→    type: (row.type as MemoryType) || 'text',
    82→    title: row.title || '',
    83→    content: row.content || '',
    84→    summary: row.summary,
    85→    tags: row.tags ? row.tags.split(',').filter(Boolean) : [],
    86→    sourceUrl: row.source_url,
    87→    fileUrl: row.file_url,
    88→    imagePreview: row.image_preview,
    89→    imageUrl: row.image_url || null,
    90→    recap: row.recap || null,
    91→    isFavorite: row.is_favorite || false,
    92→    createdAt: row.created_at,
    93→    updatedAt: row.updated_at,
    94→    collections: row.memory_collections?.map((mc) => ({
    95→      id: mc.collections.id,
    96→      name: mc.collections.name,
    97→      color: mc.collections.color,
    98→      icon: mc.collections.icon,

   114→// ─── Smart Token Saver: Local, free, instant tagging ──────────────────
   115→function getLocalTags(text: string): string[] {
   116→  const lower = text.toLowerCase()
   117→  const tags: string[] = []
   118→
   119→  // URL detection → 'link'
   120→  if (/https?:\/\//.test(lower)) {
   121→    tags.push('link')
   122→  }
   123→
   124→  // Task words → 'task'
   125→  if (/\b(todo|remind|need to|must|buy)\b/.test(lower)) {
   126→    tags.push('task')
   127→  }
   128→
   129→  // Person/personal words → 'personal'
   130→  if (/\b(i |me |with |call|email)\b/.test(lower)) {
   131→    tags.push('personal')
   132→  }
   133→
   134→  // Question/idea words → 'idea'
   135→  if (/\b(what|how|why|who)\b/.test(lower)) {
   136→    tags.push('idea')
   137→  }
   138→
   139→  return tags.slice(0, 5)
   140→}

   142→// ─── Extended keyword-based auto-tagging (richer, for AI fallback) ───
   143→function autoGenerateTags(content: string, title: string): string[] {
   144→  const text = `${title} ${content}`.toLowerCase()
   145→  const tagMap: Record<string, string[]> = {
   146→    'work': ['meeting', 'project', 'deadline', 'client', 'office', 'team', 'q1', 'q2', 'q3', 'q4', 'quarterly', 'strategy'],
   147→    'personal': ['routine', 'morning', 'exercise', 'meditation', 'journal', 'habit'],
   148→    'travel': ['trip', 'itinerary', 'flight', 'hotel', 'visit', 'tokyo', 'paris', 'destination'],
   149→    'learning': ['learn', 'study', 'course', 'tutorial', 'book', 'read', 'article'],
   150→    'code': ['code', 'programming', 'react', 'javascript', 'typescript', 'api', 'bug', 'feature', 'css', 'html', 'framework'],
   151→    'design': ['design', 'ui', 'ux', 'layout', 'color', 'font', 'figma', 'wireframe'],
   152→    'ai': ['ai', 'machine learning', 'neural', 'model', 'gpt', 'gemini', 'llm', 'chatbot'],
   153→    'recipe': ['recipe', 'cook', 'bake', 'ingredient', 'food', 'meal', 'breakfast', 'dinner'],
   154→    'idea': ['idea', 'concept', 'brainstorm', 'innovative', 'startup', 'prototype'],
   155→    'finance': ['budget', 'expense', 'invest', 'savings', 'money', 'cost'],
   156→  }
   157→
   158→  const tags: string[] = []
   159→  for (const [tag, keywords] of Object.entries(tagMap)) {
   160→    if (keywords.some(kw => text.includes(kw))) {
   161→      tags.push(tag)
   162→    }
   163→  }
   164→  return tags.slice(0, 5)
   165→}

   562→  // ── Background AI auto-tagging (non-blocking) ──────────────────────
   563→  backgroundAutoTag: async (memoryId: string, content: string, title: string) => {
   564→    try {
   565→      const res = await fetch('/api/auto-tag', {
   566→        method: 'POST',
   567→        headers: { 'Content-Type': 'application/json' },
   568→        body: JSON.stringify({ content: `${title} ${content}`.slice(0, 1000) }),
   569→      })
   570→
   571→      if (!res.ok) return
   572→
   573→      const { tags: aiTags } = await res.json()
   574→      if (!aiTags || !aiTags.length) return
   575→
   576→      const state = get()
   577→
   578→      // Update in Supabase if authenticated + ready
   579→      if (state.isAuthenticated && state.supabaseReady) {
   580→        try {
   581→          const supabase = await getSupabaseBrowser()
   582→          const { data: { session } } = await supabase.auth.getSession()
   583→          if (session) {
   584→            await supabase
   585→              .from('memories')
   586→              .update({ tags: aiTags.join(',') })
   587→              .eq('id', memoryId)
   588→          }
   589→        } catch {
   590→          // Silently fail Supabase update — local state still updates
   591→        }
   592→      } else {
   593→        // Update via Prisma API
   594→        try {
   595→          await fetch(`/api/memories/${memoryId}`, {
   596→            method: 'PATCH',
   597→            headers: { 'Content-Type': 'application/json' },
   598→            body: JSON.stringify({ tags: aiTags }),
   599→          })
   600→        } catch {
   601→          // Silently fail
   602→        }
   603→      }
   604→
   605→      // Update the local state so tags appear immediately
   606→      get().updateMemory(memoryId, { tags: aiTags })
   607→    } catch {
   608→      // Non-blocking: never fail the save flow
   609→    }
   610→  },

   743→  // ── Save Memory (Supabase-aware, optimistic UI, token-saving) ────────
   744→  saveMemory: async (data) => {
   745→    const state = get()
   746→    const { type, title, content, sourceUrl, imageUrl, tags, collectionIds } = data
   747→
   748→    // ── STEP 1: Smart Token Saver — local, free, instant tags ──────────
   749→    const fullText = `${title || ''} ${content || ''}`.trim()
   750→    const localTags = getLocalTags(fullText)
   751→    const wordCount = fullText.split(/\s+/).filter(Boolean).length
   752→
   753→    // If <20 words AND local tags found: use only local tags, skip AI
   754→    const isShortNote = wordCount < 20 && localTags.length > 0
   755→
   756→    // Merge tags: caller tags > local tags > extended auto tags
   757→    let finalTags = tags
   758→    if (!tags || tags.length === 0) {
   759→      if (localTags.length > 0) {
   760→        finalTags = localTags
   761→      } else if (content?.trim()) {
   762→        const autoTags = autoGenerateTags(content, title || '')
   763→        if (autoTags.length > 0) finalTags = autoTags
   764→      }
   765→    }
   766→
   767→    // ── Authenticated + Supabase ready: save to Supabase directly ──
   768→    if (state.isAuthenticated && state.supabaseReady) {
   769→      try {
   770→        const supabase = await getSupabaseBrowser()
   771→        const { data: { session } } = await supabase.auth.getSession()
   772→        if (!session) throw new Error('No session')
   773→
   774→        const { data: memoryRow, error } = await supabase
   775→          .from('memories')
   776→          .insert({
   777→            user_id: session.user.id,
   778→            type: type || 'text',
   779→            title: title || '',
   780→            content: content || '',
   781→            source_url: sourceUrl || null,
   782→            image_url: imageUrl || null,
   783→            tags: finalTags ? (Array.isArray(finalTags) ? finalTags.join(',') : finalTags) : '',
   784→          })
   785→          .select('*, memory_collections(collection_id, collections(id, name, color, icon))')
   786→          .single()
   787→
   788→        if (error) throw error
   789→
   790→        const memory = mapSupabaseMemory(memoryRow as SupabaseMemoryRow)
   791→
   792→        // If collectionIds provided, create junction rows
   793→        if (collectionIds?.length) {
   794→          await supabase.from('memory_collections').insert(
   795→            collectionIds.map((cid: string) => ({
   796→              memory_id: memory.id,
   797→              collection_id: cid,
   798→            }))
   799→          )
   800→        }
   801→
   802→        // Optimistic UI: add to store instantly
   803→        get().addMemory(memory)
   804→
   805→        // ── STEP 2: Deferred AI tagging — ONLY if not a short note with local tags ──
   806→        if (content?.trim() && !isShortNote) {
   807→          get().backgroundAutoTag(memory.id, content, title || '')
   808→        }
   809→
   810→        // ── STEP 3: AI summary generation (background, non-blocking) ──
   811→        if (content?.trim() && !isShortNote) {
   812→          get().backgroundAutoSummary(memory.id, content)
   813→        }
   814→
   815→        // ── STEP 4: If this is a link, read the URL content (background, non-blocking) ──
   816→        if (sourceUrl?.trim()) {
   817→          get().backgroundReadLink(memory.id, sourceUrl)
   818→        }
   819→
   820→        // ── STEP 5: Generate embedding for semantic search (background, non-blocking) ──
   821→        if (content?.trim()) {
   822→          get().backgroundGenerateEmbedding(memory.id, content)
   823→        }
   824→
   825→        return memory
   826→      } catch {
   827→        // Fall through to Prisma API
   828→      }
   829→    }
   830→
   831→    // ── Fallback: save via Prisma API ──
   832→    try {
   833→      const res = await fetch('/api/memories', {
   834→        method: 'POST',
   835→        headers: { 'Content-Type': 'application/json' },
   836→        body: JSON.stringify({
   837→          type,
   838→          title,
   839→          content,
   840→          sourceUrl,
   841→          imageUrl,
   842→          tags: finalTags,
   843→          collectionIds,
   844→        }),
   845→      })
   846→      if (res.ok) {
   847→        const memory: Memory = await res.json()
   848→        get().addMemory(memory)
   849→
   850→        // ── STEP 2: Deferred AI tagging — ONLY if not a short note with local tags ──
   851→        if (content?.trim() && !isShortNote) {
   852→          get().backgroundAutoTag(memory.id, content, title || '')
   853→        }
   854→
   855→        // ── STEP 3: AI summary generation (background, non-blocking) ──
   856→        if (content?.trim() && !isShortNote) {
   857→          get().backgroundAutoSummary(memory.id, content)
   858→        }
   859→
   860→        // ── STEP 4: If this is a link, read the URL content (background, non-blocking) ──
   861→        if (sourceUrl?.trim()) {
   862→          get().backgroundReadLink(memory.id, sourceUrl)
   863→        }
   864→
   865→        // ── STEP 5: Generate embedding for semantic search (background, non-blocking) ──
   866→        if (content?.trim()) {
   867→          get().backgroundGenerateEmbedding(memory.id, content)
   868→        }
   869→
   870→        return memory
   871→      }
   872→    } catch {
   873→      // Silent fail
   874→    }
   875→    return null
   876→  },
