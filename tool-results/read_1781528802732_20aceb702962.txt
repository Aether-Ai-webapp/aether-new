   210→function parseCognitiveJSON(responseText: string): CognitiveResult {
   211→  // Extract JSON from response (handle markdown code blocks)
   212→  const jsonMatch = responseText.match(/\{[\s\S]*\}/)
   213→  if (jsonMatch) {
   214→    try {
   215→      const parsed = JSON.parse(jsonMatch[0])
   216→      return {
   217→        summary: typeof parsed.summary === 'string' ? parsed.summary.trim() : '',
   218→        deepInsight: typeof parsed.deep_insight === 'string' ? parsed.deep_insight.trim() : '',
   219→        tags: Array.isArray(parsed.tags)
   220→          ? parsed.tags
   221→              .filter((t: unknown) => typeof t === 'string')
   222→              .map((t: string) => t.toLowerCase().trim())
   223→              .filter(Boolean)
   224→              .slice(0, 5)
   225→          : [],
   226→      }
   227→    } catch {
   228→      // JSON parse failed — continue to regex extraction
   229→    }
   230→  }
   231→
   232→  // Fallback: try to extract individual fields via regex
   233→  const summaryMatch = responseText.match(/"summary"\s*:\s*"([^"]*)"/)
   234→  const insightMatch = responseText.match(/"deep_insight"\s*:\s*"([^"]*)"/)
   235→
   236→  return {
   237→    summary: summaryMatch?.[1]?.trim() || '',
   238→    deepInsight: insightMatch?.[1]?.trim() || '',
   239→    tags: [],
   240→  }
   241→}

   264→async function matchToExistingCollections(
   265→  memoryId: string,
   266→  memoryTags: string[],
   267→  userId: string
   268→): Promise<void> {
   269→  try {
   270→    const supabase = await getSupabaseServer()
   271→
   272→    // Fetch all user's collections
   273→    const { data: userCollections, error: collError } = await supabase
   274→      .from('collections')
   275→      .select('id, name')
   276→      .eq('user_id', userId)
   277→
   278→    if (collError || !userCollections) return
   279→
   280→    for (const collection of userCollections) {
   281→      const collNameLower = collection.name.toLowerCase()
   282→
   283→      // Check if any of the memory's tags match this collection's name
   284→      const tagMatches = memoryTags.some(
   285→        (tag) =>
   286→          collNameLower.includes(tag) ||
   287→          tag.includes(collNameLower) ||
   288→          levenshteinSimilarity(collNameLower, tag) > 0.6
   289→      )
   290→
   291→      if (tagMatches) {
   292→        // Check if this junction already exists
   293→        const { data: existingJunction } = await supabase
   294→          .from('memory_collections')
   295→          .select('id')
   296→          .eq('memory_id', memoryId)
   297→          .eq('collection_id', collection.id)
   298→          .limit(1)
   299→
   300→        if (!existingJunction || existingJunction.length === 0) {
   301→          await supabase.from('memory_collections').insert({
   302→            memory_id: memoryId,
   303→            collection_id: collection.id,
   304→          })
   305→          console.log(`[capture] Auto-assigned memory ${memoryId} to collection "${collection.name}"`)
   306→        }
   307→      }
   308→    }
   309→  } catch (err) {
   310→    console.error('[capture] Collection matching error:', err instanceof Error ? err.message : 'Unknown')
   311→  }
   312→}

   314→async function checkAutoCollectionRule(userId: string): Promise<void> {
   315→  try {
   316→    const supabase = await getSupabaseServer()
   317→
   318→    // Fetch all user's memories that have tags
   319→    const { data: allMemories, error: memError } = await supabase
   320→      .from('memories')
   321→      .select('id, tags')
   322→      .eq('user_id', userId)
   323→      .neq('tags', '')
   324→
   325→    if (memError || !allMemories || allMemories.length < 10) return
   326→
   327→    // Fetch all existing junction rows to determine which memories are collected
   328→    const { data: junctions } = await supabase
   329→      .from('memory_collections')
   330→      .select('memory_id')
   331→
   332→    const collectedMemoryIds = new Set((junctions || []).map((j) => j.memory_id))
   333→
   334→    // Filter to uncollected memories only
   335→    const uncollectedMemories = allMemories.filter((m) => !collectedMemoryIds.has(m.id))
   336→
   337→    if (uncollectedMemories.length < 10) return
   338→
   339→    // Build tag frequency map from uncollected memories
   340→    const tagFrequency: Record<string, string[]> = {}
   341→    for (const mem of uncollectedMemories) {
   342→      const memTags = (mem.tags as string)
   343→        .split(',')
   344→        .map((t) => t.trim().toLowerCase())
   345→        .filter(Boolean)
   346→      for (const tag of memTags) {
   347→        if (!tagFrequency[tag]) tagFrequency[tag] = []
   348→        tagFrequency[tag].push(mem.id)
   349→      }
   350→    }
   351→
   352→    // Fetch existing collection names to avoid duplicates
   353→    const { data: existingCollections } = await supabase
   354→      .from('collections')
   355→      .select('name')
   356→      .eq('user_id', userId)
   357→
   358→    const existingNames = new Set(
   359→      (existingCollections || []).map((c) => c.name.toLowerCase())
   360→    )
   361→
   362→    const clusterColors = [
   363→      '#6D597A', '#B56576', '#355070', '#EA526F',
   364→      '#23B5D3', '#7B2D8E', '#F18F01', '#C73E1D',
   365→      '#2D936C', '#566E3D', '#8B5CF6', '#EC4899',
   366→    ]
   367→    const { count: collectionCount } = await supabase
   368→      .from('collections')
   369→      .select('*', { count: 'exact', head: true })
   370→      .eq('user_id', userId)
   371→    let colorIdx = collectionCount || 0
   372→
   373→    // Find tag clusters that hit 10+ memories
   374→    for (const [tag, memoryIds] of Object.entries(tagFrequency)) {
   375→      if (memoryIds.length >= 10) {
   376→        const collectionName = tag.charAt(0).toUpperCase() + tag.slice(1)
   377→
   378→        // Skip if a collection with this name already exists
   379→        if (existingNames.has(collectionName.toLowerCase())) continue
   380→
   381→        const color = clusterColors[colorIdx % clusterColors.length]
   382→        colorIdx++
   383→
   384→        // Create the new collection
   385→        const { data: newCollection, error: createError } = await supabase
   386→          .from('collections')
   387→          .insert({
   388→            user_id: userId,
   389→            name: collectionName,
   390→            icon: getTagIcon(tag),
   391→            color,
   392→          })
   393→          .select('id')
   394→          .single()
   395→
   396→        if (createError || !newCollection) {
   397→          console.error('[capture] Auto-collection creation failed:', createError?.message)
   398→          continue
   399→        }
   400→
   401→        // Batch-assign all matching memories to this new collection
   402→        const junctionInserts = memoryIds.slice(0, 100).map((memId) => ({
   403→          memory_id: memId,
   404→          collection_id: newCollection.id,
   405→        }))
   406→
   407→        const { error: junctionError } = await supabase
   408→          .from('memory_collections')
   409→          .insert(junctionInserts)
   410→
   411→        if (junctionError) {
   412→          console.error('[capture] Junction insert failed:', junctionError.message)
   413→        } else {
   414→          console.log(`[capture] Auto-created collection "${collectionName}" with ${memoryIds.length} memories`)
   415→          existingNames.add(collectionName.toLowerCase())
   416→        }
   417→      }
   418→    }
   419→  } catch (err) {
   420→    console.error('[capture] Auto-collection rule error:', err instanceof Error ? err.message : 'Unknown')
   421→  }
   422→}

   446→// ─── Levenshtein similarity for fuzzy tag matching ─────────────────
   447→function levenshteinSimilarity(a: string, b: string): number {
   448→  if (a === b) return 1
   449→  if (!a.length || !b.length) return 0
   450→
   451→  const matrix: number[][] = []
   452→  for (let i = 0; i <= b.length; i++) matrix[i] = [i]
   453→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j
   454→
   455→  for (let i = 1; i <= b.length; i++) {
   456→    for (let j = 1; j <= a.length; j++) {
   457→      const cost = b[i - 1] === a[j - 1] ? 0 : 1
   458→      matrix[i][j] = Math.min(
   459→        matrix[i - 1][j] + 1,
   460→        matrix[i][j - 1] + 1,
   461→        matrix[i - 1][j - 1] + cost
   462→      )
   463→    }
   464→  }
   465→
   466→  const maxLen = Math.max(a.length, b.length)
   467→  return 1 - matrix[b.length][a.length] / maxLen
   468→}

   493→    collections,
   494→  }
   495→}

   497→// ═══════════════════════════════════════════════════════════════════════
   498→// ─── POST /api/capture — THE UNIVERSAL CAPTURE ENGINE ────────────────
   499→// ═══════════════════════════════════════════════════════════════════════
   500→export async function POST(req: NextRequest) {
   501→  try {
   502→    // ── STEP 1: UNIVERSAL PAYLOAD SCANNER ─────────────────────────────
   503→    const formData = await req.formData()
   504→
   505→    const textInput = (formData.get('text') as string) || ''
   506→    const urlInput = (formData.get('url') as string) || ''
   507→    const typeOverride = (formData.get('type') as string) || ''
   508→    const audioFile = formData.get('audio') as File | null
   509→    const imageFile = formData.get('image') as File | null
   510→
   511→    const hasText = textInput.trim().length > 0
   512→    const hasUrl = urlInput.trim().length > 0
   513→    const hasAudio = audioFile !== null && audioFile.size > 0
   514→    const hasImage = imageFile !== null && imageFile.size > 0
   515→
   516→    if (!hasText && !hasUrl && !hasAudio && !hasImage) {
   517→      return NextResponse.json(
   518→        { success: false, error: 'No content provided. Send text, audio, image, or url.' },
   519→        { status: 400 }
   520→      )
   521→    }
   522→
   523→    // ── STEP 2: AUTHENTICATE USER VIA SUPABASE ───────────────────────
   524→    let supabaseUserId: string | null = null
   525→    let useSupabase = false
   526→
   527→    try {
   528→      const supabase = await getSupabaseServer()
   529→      const { data: { user: authUser }, error: authError } = await supabase.auth.getUser()
   530→      if (!authError && authUser) {
   531→        supabaseUserId = authUser.id
   532→        useSupabase = true
   533→      }
   534→    } catch {
   535→      // Supabase not configured — fall back to Prisma
   536→      useSupabase = false
   537→    }
   538→
   539→    // ── STEP 3: DYNAMIC INPUT DISPATCHING ────────────────────────────
   540→
   541→    // 3a: Transcribe audio if present (z-ai ASR → Groq Whisper)
   542→    let audioTranscript = ''
   543→    if (hasAudio && audioFile) {
   544→      audioTranscript = await transcribeAudio(audioFile)
   545→    }
   546→
   547→    // 3b: Upload image to Supabase Storage if present
   548→    let imageUrl: string | null = null
   549→    let imagePreviewUrl: string | null = null
   550→    if (hasImage && imageFile) {
   551→      if (useSupabase && supabaseUserId) {
   552→        imageUrl = await uploadImageToStorage(imageFile, supabaseUserId)
   553→      }
   554→      if (!imageUrl) {
   555→        // Fallback: base64 preview if storage upload fails or no Supabase
   556→        try {
   557→          const arrayBuffer = await imageFile.arrayBuffer()
   558→          const base64 = Buffer.from(arrayBuffer).toString('base64')
   559→          const mimeType = imageFile.type || 'image/png'
   560→          imagePreviewUrl = `data:${mimeType};base64,${base64}`
   561→        } catch {
   562→          imagePreviewUrl = null
   563→        }
   564→      }
   565→    }
   566→
   567→    // 3c: Determine content type
   568→    let memoryType = 'text'
   569→    if (typeOverride && ['text', 'voice', 'link', 'image'].includes(typeOverride)) {
   570→      memoryType = typeOverride
   571→    } else if (hasImage) {
   572→      memoryType = 'image'
   573→    } else if (hasAudio && !hasText && !hasUrl) {
   574→      memoryType = 'voice'
   575→    } else if (hasUrl || (hasText && /https?:\/\//.test(textInput.toLowerCase()))) {
   576→      memoryType = 'link'
   577→    }
   578→
   579→    // 3d: Compose the raw content string
   580→    let rawContent = ''
   581→    if (audioTranscript.trim()) {
   582→      rawContent = audioTranscript.trim()
   583→      if (hasText && textInput.trim()) {
   584→        rawContent = `${textInput.trim()}\n\n${audioTranscript.trim()}`
   585→      }
   586→    } else if (hasText) {
   587→      rawContent = textInput.trim()
   588→    }
   589→
   590→    const sourceUrl = hasUrl ? urlInput.trim() : memoryType === 'link' ? textInput.trim() : null
   591→
   592→    if (hasImage && !rawContent.trim()) {
   593→      rawContent = imageFile ? `Image: ${imageFile.name}` : 'Image'
   594→    }
   595→
   596→    // Compose the title
   597→    let title = rawContent.length > 80 ? rawContent.slice(0, 80) + '...' : rawContent
   598→    if (hasImage && !textInput.trim()) {
   599→      title = imageFile ? `Image: ${imageFile.name}` : 'Image'
   600→    }
   601→
   602→    // ── STEP 4: COGNITIVE SYNTHESIS — 3-TIER AI PIPELINE ───────────
   603→    const textForSynthesis = rawContent || (hasUrl ? urlInput : '')
   604→    let cleanedSummary: string | null = null
   605→    let deepInsight: string | null = null
   606→    let aiTags: string[] = []
   607→
   608→    // Always run keyword tags first (instant, free, reliable)
   609→    const keywordTags = autoGenerateTags(rawContent, title)
   610→
   611→    // Run AI synthesis if we have content (with 10-second timeout)
   612→    if (textForSynthesis.trim()) {
   613→      try {
   614→        const synthesisPromise = generateCognitiveSynthesis(textForSynthesis)
   615→        const timeoutPromise = new Promise<CognitiveResult>((resolve) =>
   616→          setTimeout(() => resolve({ summary: '', deepInsight: '', tags: [] }), 10000)
   617→        )
   618→
   619→        const cognitive = await Promise.race([synthesisPromise, timeoutPromise])
   620→        if (cognitive.summary) cleanedSummary = cognitive.summary
   621→        if (cognitive.deepInsight) deepInsight = cognitive.deepInsight
   622→        if (cognitive.tags.length > 0) aiTags = cognitive.tags
   623→      } catch (err) {
   624→        console.error('[capture] Synthesis error:', err instanceof Error ? err.message : 'Unknown')
   625→      }
   626→    }
   627→
   628→    // Merge: AI tags take priority, keyword tags fill gaps
   629→    const finalTags = aiTags.length > 0
   630→      ? [...new Set([...aiTags, ...keywordTags])].slice(0, 5)
   631→      : keywordTags
   632→
   633→    // ── STEP 5: DATABASE INGESTION ───────────────────────────────────
   634→    let memory: Record<string, unknown>
   635→
   636→    // 5a: Supabase-first insertion (authenticated users with RLS)
   637→    if (useSupabase && supabaseUserId) {
   638→      try {
   639→        const supabase = await getSupabaseServer()
   640→        const insertData: Record<string, unknown> = {
   641→          user_id: supabaseUserId,
   642→          type: memoryType,
   643→          title,
   644→          content: rawContent,
   645→          source_url: sourceUrl || null,
   646→          tags: finalTags.join(','),
   647→        }
   648→
   649→        if (cleanedSummary) insertData.summary = cleanedSummary
   650→        if (deepInsight) insertData.deep_insight = deepInsight
   651→        if (imageUrl) insertData.image_url = imageUrl
   652→        if (imagePreviewUrl && !imageUrl) insertData.image_preview = imagePreviewUrl
   653→
   654→        const { data: memoryRow, error: insertError } = await supabase
   655→          .from('memories')
   656→          .insert(insertData)
   657→          .select('*, memory_collections(collection_id, collections(id, name, color, icon))')
   658→          .single()
   659→
   660→        if (insertError) throw insertError
   661→
   662→        const row = memoryRow as Record<string, unknown>
   663→        const collectionsFromRow = (
   664→          (row.memory_collections as Record<string, Record<string, unknown>>[]) || []
   665→        ).map((mc) => ({
   666→          id: (mc.collections as Record<string, unknown>)?.id as string,
   667→          name: (mc.collections as Record<string, unknown>)?.name as string,
   668→          color: (mc.collections as Record<string, unknown>)?.color as string,
   669→          icon: (mc.collections as Record<string, unknown>)?.icon as string,
   670→        }))
   671→
   672→        memory = buildMemoryResponse(row, collectionsFromRow)
   673→
   674→        // Background: embedding + link reading + auto-collections
   675→        if (rawContent.trim()) {
   676→          const baseUrl = new URL(req.url).origin
   677→          fetch(`${baseUrl}/api/generate-embedding`, {
   678→            method: 'POST',
   679→            headers: { 'Content-Type': 'application/json' },
   680→            body: JSON.stringify({ memoryId: memory.id, content: rawContent.slice(0, 2000) }),
   681→          }).catch(() => {})
   682→        }
   683→
   684→        if (sourceUrl?.trim()) {
   685→          const baseUrl = new URL(req.url).origin
   686→          fetch(`${baseUrl}/api/ai/read-link`, {
   687→            method: 'POST',
   688→            headers: { 'Content-Type': 'application/json' },
   689→            body: JSON.stringify({ url: sourceUrl }),
   690→          }).catch(() => {})
   691→        }
   692→
   693→        runAutonomousCollections(memory.id as string, finalTags, supabaseUserId).catch(() => {})
   694→
   695→        return NextResponse.json({ success: true, memory })
   696→      } catch (err) {
   697→        console.error('[capture] Supabase insertion failed, falling back to Prisma:', err instanceof Error ? err.message : 'Unknown')
   698→      }
   699→    }
   700→
   701→    // 5b: Fallback — Prisma SQLite insertion (local / no Supabase)
   702→    try {
   703→      const { db } = await import('@/lib/db')
   704→
   705→      const prismaMemory = await db.memory.create({
   706→        data: {
   707→          type: memoryType,
   708→          title,
   709→          content: rawContent,
   710→          summary: cleanedSummary,
   711→          deepInsight,
   712→          sourceUrl: sourceUrl || null,
   713→          tags: finalTags.join(','),
   714→          imageUrl: imageUrl || null,
   715→          imagePreview: imagePreviewUrl && !imageUrl ? imagePreviewUrl : null,
   716→        },
   717→        include: {
   718→          collections: {
   719→            include: {
   720→              collection: {
   721→                select: { id: true, name: true, color: true, icon: true },
   722→              },
   723→            },
   724→          },
   725→        },
   726→      })
   727→
   728→      memory = buildMemoryResponse(
   729→        {
   730→          id: prismaMemory.id,
   731→          type: prismaMemory.type,
   732→          title: prismaMemory.title,
   733→          content: prismaMemory.content,
   734→          summary: prismaMemory.summary,
   735→          deepInsight: prismaMemory.deepInsight,
   736→          tags: prismaMemory.tags,
   737→          sourceUrl: prismaMemory.sourceUrl,
   738→          fileUrl: prismaMemory.fileUrl,
   739→          imagePreview: prismaMemory.imagePreview,
   740→          imageUrl: prismaMemory.imageUrl,
   741→          isFavorite: prismaMemory.isFavorite,
   742→          created_at: prismaMemory.createdAt.toISOString(),
   743→          updated_at: prismaMemory.updatedAt.toISOString(),
   744→        },
   745→        prismaMemory.collections.map((mc) => ({
   746→          id: mc.collection.id,
   747→          name: mc.collection.name,
   748→          color: mc.collection.color,
   749→          icon: mc.collection.icon,
   750→        }))
   751→      )
   752→
   753→      // Background: embedding + link reading
   754→      if (rawContent.trim()) {
   755→        const baseUrl = new URL(req.url).origin
   756→        fetch(`${baseUrl}/api/generate-embedding`, {
   757→          method: 'POST',
   758→          headers: { 'Content-Type': 'application/json' },
   759→          body: JSON.stringify({ memoryId: prismaMemory.id, content: rawContent.slice(0, 2000) }),
   760→        }).catch(() => {})
   761→      }
   762→
   763→      if (sourceUrl?.trim()) {
   764→        const baseUrl = new URL(req.url).origin
   765→        fetch(`${baseUrl}/api/ai/read-link`, {
   766→          method: 'POST',
   767→          headers: { 'Content-Type': 'application/json' },
   768→          body: JSON.stringify({ url: sourceUrl }),
   769→        }).catch(() => {})
   770→      }
   771→
   772→      return NextResponse.json({ success: true, memory })
   773→    } catch (err) {
   774→      console.error('[capture] Prisma insertion failed:', err instanceof Error ? err.message : 'Unknown')
   775→      return NextResponse.json(
   776→        { success: false, error: 'Failed to save memory to database' },
   777→        { status: 500 }
   778→      )
   779→    }
   780→
   781→  } catch (error) {
   782→    console.error('[capture] Unhandled error:', error instanceof Error ? error.message : 'Unknown')
   783→    return NextResponse.json(
   784→      { success: false, error: 'Internal server error during capture' },
   785→      { status: 500 }
   786→    )
   787→  }
   788→}

   790→// ═══════════════════════════════════════════════════════════════════════
   791→// ─── DELETE /api/capture — PURGE MEMORY FROM SUPABASE ────────────────
   792→// ═══════════════════════════════════════════════════════════════════════
   793→export async function DELETE(req: NextRequest) {
   794→  try {
   795→    const { searchParams } = new URL(req.url)
   796→    const memoryId = searchParams.get('id')
   797→
   798→    if (!memoryId) {
   799→      return NextResponse.json(
   800→        { success: false, error: 'Memory ID is required' },
   801→        { status: 400 }
   802→      )
   803→    }
   804→
   805→    // Try Supabase deletion first
   806→    let supabaseDeleted = false
   807→    try {
   808→      const supabase = await getSupabaseServer()
   809→      const { data: { user: authUser }, error: authError } = await supabase.auth.getUser()
   810→
   811→      if (!authError && authUser) {
   812→        // Delete junction rows first (foreign key constraint)
   813→        const { error: junctionError } = await supabase
   814→          .from('memory_collections')
   815→          .delete()
   816→          .eq('memory_id', memoryId)
   817→
   818→        if (junctionError) {
   819→          console.error('[capture] Junction delete error:', junctionError.message)
   820→        }
   821→
   822→        // Delete the memory itself
   823→        const { error: deleteError } = await supabase
   824→          .from('memories')
   825→          .delete()
   826→          .eq('id', memoryId)
   827→          .eq('user_id', authUser.id)
   828→
   829→        if (!deleteError) {
   830→          supabaseDeleted = true
   831→        } else {
   832→          console.error('[capture] Supabase delete failed:', deleteError.message)
   833→        }
   834→      }
   835→    } catch {
   836→      // Supabase not available — fall through to Prisma
   837→    }
   838→
   839→    // Fallback: Prisma deletion (always attempt if Supabase didn't succeed)
   840→    if (!supabaseDeleted) {
   841→      try {
   842→        const { db } = await import('@/lib/db')
   843→        await db.memory.delete({ where: { id: memoryId } })
   844→      } catch (prismaErr) {
   845→        console.error('[capture] Prisma delete failed:', prismaErr instanceof Error ? prismaErr.message : 'Unknown')
   846→        // If both failed, return error
   847→        if (!supabaseDeleted) {
   848→          return NextResponse.json(
   849→            { success: false, error: 'Failed to delete memory' },
   850→            { status: 500 }
   851→          )
   852→        }
   853→      }
   854→    }
   855→
   856→    return NextResponse.json({ success: true })
   857→
   858→  } catch (error) {
   859→    console.error('[capture] Delete error:', error instanceof Error ? error.message : 'Unknown')
   860→    return NextResponse.json(
   861→      { success: false, error: 'Internal server error during deletion' },
   862→      { status: 500 }
   863→    )
   864→  }
   865→}
