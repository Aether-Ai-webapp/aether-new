   336→   336→   336→    for (const mem of uncollectedMemories) {

   338→   338→   338→      for (const tag of memTags) {

   429→   429→   429→// ─── Levenshtein similarity for fuzzy tag matching ─────────────────
   430→   430→   430→function levenshteinSimilarity(a: string, b: string): number {
   431→   431→   431→  if (a === b) return 1
   432→   432→   432→  if (!a.length || !b.length) return 0
   433→   433→   433→
   434→   434→   434→  const matrix: number[][] = []
   435→   435→   435→  for (let i = 0; i <= b.length; i++) matrix[i] = [i]
   436→   436→   436→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j
   437→   437→   437→
   438→   438→   438→  for (let i = 1; i <= b.length; i++) {
   439→   439→   439→    for (let j = 1; j <= a.length; j++) {
   440→   440→   440→      const cost = b[i - 1] === a[j - 1] ? 0 : 1
   441→   441→   441→      matrix[i][j] = Math.min(
   442→   442→   442→        matrix[i - 1][j] + 1,
   443→   443→   443→        matrix[i][j - 1] + 1,
   444→   444→   444→        matrix[i - 1][j - 1] + cost
   445→   445→   445→      )
   446→   446→   446→    }
   447→   447→   447→  }
   448→   448→   448→
   449→   449→   449→  const maxLen = Math.max(a.length, b.length)
   450→   450→   450→  return 1 - matrix[b.length][a.length] / maxLen
   451→   451→   451→}
